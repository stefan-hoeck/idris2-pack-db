 25 |            Just ty <- lookupTyExact n (gamma defs)
 26 |                | Nothing => undefinedName fc n
 27 |            pure $ gnf env (embed ty)
 28 |   chk env (Meta fc n i args)

 30 |            Just mty <- lookupTyExact (Resolved i) (gamma defs)

 33 |   chk env (Bind fc nm b sc)

 36 |            pure $ gnf env (discharge fc nm b !(getTerm bt) !(getTerm sct))
 37 |   chk env (App fc f a)

 39 |            case !(getNF fty) of
 40 |                 NBind _ _ (Pi _ _ _ ty) scdone =>
 41 |                       do defs <- get Ctxt
 42 |                          aty <- chk env a
 43 |                          sc' <- scdone defs (toClosure defaultOpts env a)
 44 |                          pure $ glueBack defs env sc'
