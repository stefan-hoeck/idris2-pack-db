0 | {-
  1 | Optimisations which rewrite constructors and cases to more optimised versions.
  2 | In general optimisations here may change the representation of data types
  3 | so after making changes here you may need to update the ttc version
  4 |
  5 | Nat hack - constructors with ZERO or SUCC ConInfo are replaced with 0 : Integer and (1 +) : Integer -> Integer respectively
  6 |          - cases on ZERO and SUCC are replaced with checking if the integer is 0
  7 |
  8 | Enums - enums (ie data types where all constructors are nullary after erasure) are replaced with a finite integer (Bits8, Bits16 or Bits32)
  9 |
 10 | Unit - case blocks scrutinising UNIT are removed (possibly leaving behind a non-inlinable let binding)
 11 |
 12 | Intrinsic constructors: NIL/CONS, NOTHING/JUST - these are left as constructors,
 13 |     but the name and tag are based on the ConInfo rather than the original name and tag.
 14 |     new `CDef`s must be generated for each of these constructors - see intrinsicCons.
 15 | -}

 36 | -- Rewrite applications of Nat-like constructors and functions to more optimal
 37 | -- versions using Integer
 38 |
 39 | -- None of these should be hard coded, but we'll do it this way until we
 40 | -- have a more general approach to optimising data types!
 41 | -- NOTE: Make sure that names mentioned here are listed in 'natHackNames' in
 42 | -- Common.idr, so that they get compiled, as they won't be spotted by the
 43 | -- usual calls to 'getRefs'.
 44 | data Magic : Type where
 45 |     MagicCRef :
 46 |         Name -> (arity : Nat) -> -- checks
 47 |         (FC -> FC -> forall vars. Vect arity (CExp vars) -> CExp vars) -> --translation
 48 |         Magic
 49 |
 50 | magic : List Magic -> CExp vars -> Maybe (CExp vars)
 51 | magic ms (CLam fc x exp) = CLam fc x <$> magic ms exp
 52 | magic ms e = go ms e where
 53 |
 54 |     fire : Magic -> CExp vars -> Maybe (CExp vars)
 55 |     fire (MagicCRef n arity f) (CApp fc (CRef fc' n') es) = do
 56 |         guard (n == n')
 57 |         map (f fc fc') (toVect arity es)
 58 |     fire _ _ = Nothing
 59 |
 60 |     go : List Magic -> CExp vars -> Maybe (CExp vars)
 61 |     go [] e = Nothing
 62 |     go (m :: ms) e = case fire m e of
 63 |         Nothing => go ms e
 64 |         Just e' => Just e'
 65 |
 66 | magic__integerToNat : FC -> FC -> forall vars. Vect 1 (CExp vars) -> CExp vars
 67 | magic__integerToNat fc fc' [k]
 68 |   = CApp fc (CRef fc' (NS typesNS (UN $ Basic "prim__integerToNat"))) [k]
 69 |
 70 | magic__natMinus : FC -> FC -> forall vars. Vect 2 (CExp vars) -> CExp vars
 71 | magic__natMinus fc fc' [m, n]
 72 |   = magic__integerToNat fc fc'
 73 |     [COp fc (Sub IntegerType) [m, n]]
 74 |
 75 | -- We don't reuse natMinus here because we assume that unsuc will only be called
 76 | -- on S-headed numbers so we do not need the truncating integerToNat call!
 77 | magic__natUnsuc : FC -> FC -> forall vars. Vect 1 (CExp vars) -> CExp vars
 78 | magic__natUnsuc fc fc' [m]
 79 |   = COp fc (Sub IntegerType) [m, CPrimVal fc (BI 1)]
 80 |
 81 | -- TODO: next release remove this and use %builtin pragma
 82 | natHack : List Magic
 83 | natHack =
 84 |     [ MagicCRef (NS typesNS (UN $ Basic "natToInteger")) 1 (\ _, _, [k] => k)
 85 |     , MagicCRef (NS typesNS (UN $ Basic "integerToNat")) 1 magic__integerToNat
 86 |     , MagicCRef (NS typesNS (UN $ Basic "plus")) 2
 87 |          (\ fc, fc', [m,n] => COp fc (Add IntegerType) [m, n])
 88 |     , MagicCRef (NS typesNS (UN $ Basic "mult")) 2
 89 |          (\ fc, fc', [m,n] => COp fc (Mul IntegerType) [m, n])
 90 |     , MagicCRef (NS typesNS (UN $ Basic "minus")) 2 magic__natMinus
 91 |     , MagicCRef (NS typesNS (UN $ Basic "equalNat")) 2
 92 |          (\ fc, fc', [m,n] => COp fc (EQ IntegerType) [m, n])
 93 |     , MagicCRef (NS typesNS (UN $ Basic "compareNat")) 2
 94 |          (\ fc, fc', [m,n] => CApp fc (CRef fc' (NS eqOrdNS (UN $ Basic "compareInteger"))) [m, n])
 95 |     ]

 99 | builtinMagic = magic natHack

101 | natBranch :  CConAlt vars -> Bool
102 | natBranch (MkConAlt n ZERO _ _ _) = True
103 | natBranch (MkConAlt n SUCC _ _ _) = True
104 | natBranch _ = False
105 |
106 | trySBranch : CExp vars -> CConAlt vars -> Maybe (CExp vars)
107 | trySBranch n (MkConAlt nm SUCC _ [arg] sc)
108 |     = Just (CLet (getFC n) arg YesInline (magic__natUnsuc (getFC n) (getFC n) [n]) sc)
109 | trySBranch _ _ = Nothing
110 |
111 | tryZBranch : CConAlt vars -> Maybe (CExp vars)
112 | tryZBranch (MkConAlt n ZERO _ [] sc) = Just sc
113 | tryZBranch _ = Nothing
114 |
115 | getSBranch : CExp vars -> List (CConAlt vars) -> Maybe (CExp vars)
116 | getSBranch n [] = Nothing
117 | getSBranch n (x :: xs) = trySBranch n x <|> getSBranch n xs
118 |
119 | getZBranch : List (CConAlt vars) -> Maybe (CExp vars)
120 | getZBranch [] = Nothing
121 | getZBranch (x :: xs) = tryZBranch x <|> getZBranch xs
122 |
123 | -- Rewrite case trees on Nat to be case trees on Integer
124 | nat : {auto s : Ref NextMN Int} -> CExp vars -> Core (Maybe (CExp vars))
125 | nat (CCon fc _ ZERO _ []) = pure $ Just $ CPrimVal fc (BI 0)
126 | nat (CCon fc _ SUCC _ [x]) = pure $ Just $ COp fc (Add IntegerType) [CPrimVal fc (BI 1), x]
127 | nat (CConCase fc sc@(CLocal _ _) alts def) =
128 |     pure $ if any natBranch alts
129 |             then
130 |                 let defb = fromMaybe (CCrash fc "Nat case not covered") def
131 |                     salt = maybe defb id (getSBranch sc alts)
132 |                     zalt = maybe defb id (getZBranch alts)
133 |                 in Just (CConstCase fc sc [MkConstAlt (BI 0) zalt] (Just salt))
134 |             else Nothing
135 | nat (CConCase fc sc alts def) = do
136 |     x <- newMN "succ"
137 |     Just t <- nat (CConCase fc (CLocal fc First) (map weaken alts) (map weaken def))
138 |         | Nothing => pure Nothing
139 |     pure $ Just $ CLet fc x YesInline sc t
140 | nat _ = pure Nothing

148 | enumTag : Nat -> Int -> Constant
149 | enumTag k i =
150 |   if      k <= 0xff   then B8 (cast i)
151 |   else if k <= 0xffff then B16 (cast i)
152 |   else                     B32 (cast i)

154 | enum : CExp vars -> Maybe (CExp vars)
155 | enum (CCon fc _ (ENUM n) (Just tag) []) = Just (CPrimVal fc (enumTag n tag))
156 | enum (CConCase fc sc alts def) = do
157 |     alts' <- traverse toEnum alts
158 |     Just $ CConstCase fc sc alts' def
159 |   where
160 |     toEnum : CConAlt vars -> Maybe (CConstAlt vars)
161 |     toEnum (MkConAlt nm (ENUM n) (Just tag) [] sc)
162 |         = pure $ MkConstAlt (enumTag n tag) sc
163 |     toEnum _ = Nothing
164 | enum t = Nothing

174 |     let [MkConAlt _ UNIT _ [] e] = alts

202 | intrinsicCons = process
203 |     [ (NIL, 0)

213 | tryIntrinsic : CExp vars -> Maybe (CExp vars)

225 | parameters (try : forall vars. CExp vars -> Core (CExp vars))
226 |     rewriteCExp : CExp vars -> Core (CExp vars)
227 |     -- Recursively rewrite the sub-terms of a CExp
228 |     rewriteSub : CExp vars -> Core (CExp vars)
229 |
230 |     rewriteCConAlt : CConAlt vars -> Core (CConAlt vars)
231 |     rewriteCConstAlt : CConstAlt vars -> Core (CConstAlt vars)
232 |
233 |     rewriteCExp exp = do
234 |         exp' <- rewriteSub exp
235 |         try exp'
236 |
237 |     rewriteSub (CLam fc x e) =
238 |         CLam fc x <$> rewriteCExp e
239 |     rewriteSub (CLet fc x inl e1 e2) =
240 |         CLet fc x inl <$> rewriteCExp e1 <*> rewriteCExp e2
241 |     rewriteSub (CApp fc f xs) =
242 |         CApp fc <$> rewriteCExp f <*> traverse rewriteCExp xs
243 |     rewriteSub (CCon fc n ci t xs) =
244 |         CCon fc n ci t <$> traverse rewriteCExp xs
245 |     rewriteSub (COp fc op xs) =
246 |         COp fc op <$> traverseVect rewriteCExp xs
247 |     rewriteSub (CExtPrim fc n xs) =
248 |         CExtPrim fc n <$> traverse rewriteCExp xs
249 |     rewriteSub (CForce fc lr e) =
250 |         CForce fc lr <$> rewriteCExp e
251 |     rewriteSub (CDelay fc lr e) =
252 |         CDelay fc lr <$> rewriteCExp e
253 |     rewriteSub (CConCase fc x alts def) =
254 |         CConCase fc
255 |             <$> rewriteCExp x
256 |             <*> traverse rewriteCConAlt alts
257 |             <*> traverseOpt rewriteCExp def
258 |     rewriteSub (CConstCase fc x alts def) =
259 |         CConstCase fc
260 |             <$> rewriteCExp x
261 |             <*> traverse rewriteCConstAlt alts
262 |             <*> traverseOpt rewriteCExp def
263 |     rewriteSub e = pure e
264 |
265 |     rewriteCConAlt (MkConAlt n ci t as e) = MkConAlt n ci t as <$> rewriteCExp e
266 |     rewriteCConstAlt (MkConstAlt x e) = MkConstAlt x <$> rewriteCExp e
267 |
268 | sequence : List (forall vars. CExp vars -> Core (Maybe (CExp vars))) -> CExp vars -> Core (CExp vars)
269 | sequence [] e = pure e
270 | sequence (x :: xs) e = do
271 |     e' <- fromMaybe e <$> x e
272 |     sequence xs e'
