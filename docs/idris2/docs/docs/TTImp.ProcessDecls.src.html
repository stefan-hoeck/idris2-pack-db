 41 | processFailing :
 42 |   {vars : _} ->
 43 |   {auto c : Ref Ctxt Defs} ->
 44 |   {auto m : Ref MD Metadata} ->
 45 |   {auto u : Ref UST UState} ->
 46 |   {auto s : Ref Syn SyntaxInfo} ->
 47 |   {auto o : Ref ROpts REPLOpts} ->
 48 |   List ElabOpt ->
 49 |   NestedNames vars -> Env Term vars ->
 50 |   FC -> Maybe String ->  List ImpDecl -> Core ()
 51 | processFailing eopts nest env fc mmsg decls
 52 |     = do -- save the state: the content of a failing block should be discarded
 53 |          ust <- get UST
 54 |          syn <- get Syn
 55 |          md <- get MD
 56 |
 57 |          -- We expect the block to fail and so the definitions introduced
 58 |          -- in it should be discarded once we leave the block.
 59 |          defs <- branch
 60 |
 61 |          -- We're going to run the elaboration, and then return:
 62 |          -- * Nothing     if the block correctly failed
 63 |          -- * Just err    if it either did not fail or failed with an invalid error message
 64 |          result <- catch
 65 |                (do -- Run the elaborator
 66 |                    before <- getTotalityErrors
 67 |                    traverse_ (processDecl eopts nest env) decls
 68 |                    after <- getTotalityErrors
 69 |                    let errs = after \\ before
 70 |                    let (e :: es) = errs
 71 |                      | [] => do -- We better have unsolved holes
 72 |                                 -- checkUserHoles True -- do we need this one too?
 73 |
 74 |                                  -- should we only look at the ones introduced in the block?
 75 |                                 Nothing <- checkDelayedHoles
 76 |                                   | Just err => throw err
 77 |
 78 |                                 -- Or we have (unfortunately) succeeded
 79 |                                 pure (Just $ FailingDidNotFail fc)
 80 |                    let Just msg = mmsg
 81 |                      | _ => pure Nothing
 82 |                    log "elab.failing" 10 $ "Failing block based on \{show msg} failed with \{show errs}"
 83 |                    test <- anyM (checkError msg) errs
 84 |                    pure $ do -- Unless the error is the expected one
 85 |                              guard (not test)
 86 |                              -- We should complain we had the wrong one
 87 |                              pure (FailingWrongError fc msg (e ::: es)))
 88 |                (\err => do let Just msg = mmsg
 89 |                                  | _ => pure Nothing
 90 |                            log "elab.failing" 10 $ "Failing block based on \{show msg} failed with \{show err}"
 91 |                            test <- checkError msg err
 92 |                            pure $ do -- Unless the error is the expected one
 93 |                                      guard (not test)
 94 |                                      -- We should complain we had the wrong one
 95 |                                      pure (FailingWrongError fc msg (err ::: [])))
 96 |          md' <- get MD
 97 |          -- Reset the state
 98 |          put UST ust
 99 |          put Syn syn
100 |          -- For metadata, we preserve the syntax highlithing information (but none
101 |          -- of the things that may include code that's dropped like types, LHSs, etc.)
102 |          put MD ({ semanticHighlighting := semanticHighlighting md'
103 |                  , semanticAliases := semanticAliases md'
104 |                  , semanticDefaults := semanticDefaults md'
105 |                  } md)
106 |          put Ctxt defs
107 |          -- And fail if the block was successfully accepted
108 |          whenJust result throw

203 | processTTImpFile fname
204 |     = do modIdent <- ctxtPathToNS fname
205 |          Right (ws, decor, tti) <- logTime 0 "Parsing" $
206 |                             coreLift $ parseFile fname (PhysicalIdrSrc modIdent)
207 |                             (do decls <- prog (PhysicalIdrSrc modIdent)
208 |                                 eoi
209 |                                 pure decls)
210 |                | Left err => do coreLift (putStrLn (show err))
211 |                                 pure False
212 |          traverse_ recordWarning ws
213 |          logTime 0 "Elaboration" $
214 |             catch (do ignore $ processTTImpDecls (MkNested []) Env.empty tti
215 |                       Nothing <- checkDelayedHoles
216 |                           | Just err => throw err
