 51 |             {auto o : Ref ROpts REPLOpts} ->
 52 |             Core ()
 53 | updateEnv
 54 |     = do defs <- get Ctxt
 55 |          noColor <- coreLift [ isJust noc || not tty | noc <- idrisGetEnv "NO_COLOR", tty <- isTTY stdout ]
 56 |          when noColor $ setColor False
 57 |          bprefix <- coreLift $ idrisGetEnv "IDRIS2_PREFIX"
 58 |          setPrefix (fromMaybe yprefix bprefix)
 59 |          bpath <- coreLift $ idrisGetEnv "IDRIS2_PATH"
 60 |          whenJust bpath $ traverseList1_ addExtraDir . splitPaths
 61 |          bdata <- coreLift $ idrisGetEnv "IDRIS2_DATA"
 62 |          whenJust bdata $ traverseList1_ addDataDir . splitPaths
 63 |          blibs <- coreLift $ idrisGetEnv "IDRIS2_LIBS"
 64 |          whenJust blibs $ traverseList1_ addLibDir . splitPaths
 65 |          pdirs <- coreLift $ idrisGetEnv "IDRIS2_PACKAGE_PATH"
 66 |          whenJust pdirs $ traverseList1_ addPackageSearchPath . splitPaths
 67 |          cg <- coreLift $ idrisGetEnv "IDRIS2_CG"
 68 |          whenJust cg $ \ e => case getCG (options defs) e of
 69 |            Just cg => setCG cg
 70 |            Nothing => throw (InternalError ("Unknown code generator " ++ show e))
 71 |          inccgs <- coreLift $ idrisGetEnv "IDRIS2_INC_CGS"
 72 |          whenJust inccgs $ \ cgs =>
 73 |            traverseList1_ (setIncrementalCG False) $
 74 |              map trim (split (==',') cgs)
 75 |          -- IDRIS2_PATH goes first so that it overrides this if there's
 76 |          -- any conflicts. In particular, that means that setting IDRIS2_PATH
 77 |          -- for the tests means they test the local version not the installed
 78 |          -- version
 79 |          defs <- get Ctxt
 80 |          -- add global package path to the package search paths (after those
 81 |          -- added by the user with IDRIS2_PACKAGE_PATH)

 83 |          -- These might fail while bootstrapping
 84 |          catch (addPkgDir "prelude" anyBounds) (const (pure ()))
 85 |          catch (addPkgDir "base" anyBounds) (const (pure ()))
 86 |          addDataDir (prefix_dir (dirs (options defs)) </>
 87 |                         ("idris2-" ++ showVersion False version) </> "support")
