 21 | -- The Size-Change Principle for Termination
 22 | -- by Chin Soon Lee, Neil D. Jones, Amir M. Ben-Amram

 25 | ------------------------------------------------------------------------
 26 | -- Basic types
 27 |
 28 | ||| Refer to an argument by its position
 29 | -- This seems to assume that a function has a set number of arguments
 30 | -- (a constraint we currently enforce: all clauses' LHS need to have
 31 | -- the same number of arguments)
 32 | Arg : Type
 33 | Arg = Nat
 34 |
 35 | ||| A change in (g y₀ ⋯ yₙ) with respect to (f x₀ ⋯ xₙ) is
 36 | ||| for argument yᵢ in g and xⱼ in f:
 37 | |||   Smaller        if yᵢ < xⱼ
 38 | |||   Equal          if yᵢ <= xⱼ
 39 | |||   Unknown        if yᵢ is not related to xⱼ
 40 | Change : {- f g -} Type
 41 | Change = Matrix SizeChange
 42 |
 43 | ||| A sequence in the call graph lists the source location of the successive
 44 | ||| calls and the name of each of the functions being called
 45 | ||| TODO: also record the arguments so that we can print e.g.
 46 | |||   flip x y -> flop y x -> flip x y
 47 | ||| instead of the less useful
 48 | |||   flip -> flop -> flip
 49 | public export
 50 | CallSeq : {- f g -} Type
 51 | CallSeq = List (FC, Name)
 52 |
 53 | ||| An edge from f to g is given by:
 54 | |||   the actual changes in g with respect to f
 55 | |||   the sequence in the call graph leading from f to g
 56 | ||| The path is only here for error-reporting purposes
 57 | record Graph {- f g -} where
 58 |   constructor MkGraph
 59 |   change : Change {- f g -}
 60 |   seq : CallSeq {- f g -}
 61 |
 62 | ||| Sc graphs to be added
 63 | WorkList : Type
 64 | WorkList = SortedSet (Name, Name, Graph)
 65 |
 66 | ||| Transitively-closed (modulo work list) set of sc-graphs
 67 | ||| Note: oh if only we had dependent name maps!
 68 | SCSet : Type
 69 | SCSet = NameMap {- \ f => -}
 70 |       $ NameMap {- \ g => -}
 71 |       $ SortedSet $ Graph {- f g -}

 74 | -- Instances
 75 |
 76 | -- ignore call sequence
 77 | Eq Graph where
 78 |   (==) a1 a2 = a1.change == a2.change
 79 |
 80 | Ord Graph where
 81 |   compare a1 a2 = compare a1.change a2.change

 83 | Show Graph where
 84 |   show a = show a.change ++ "\n    via call seq: " ++ show a.seq ++ "\n"

 89 | ||| Empty set of sc-graphs
 90 | initSCSet : SCSet
 91 | initSCSet = empty
 92 |
 93 | lookupMap : Name -> NameMap (NameMap v) -> NameMap v
 94 | lookupMap n m = fromMaybe empty (lookup n m)
 95 |
 96 | lookupSet : Ord v => Name -> NameMap (SortedSet v) -> SortedSet v
 97 | lookupSet n m = fromMaybe empty (lookup n m)
 98 |
 99 | lookupGraphs : Name -> Name -> SCSet -> SortedSet Graph
100 | lookupGraphs f g = lookupSet g . lookupMap f
101 |
102 | ||| Smart filter: only keep the paths starting from f
103 | selectDom : Name -> SCSet -> SCSet
104 | selectDom f s = insert f (lookupMap f s) empty
105 |
106 | ||| Smart filter: only keep the paths ending in g
107 | selectCod : Name -> SCSet -> SCSet
108 | selectCod g s = map (\m => insert g (lookupSet g m) empty) s
109 |
110 | foldl : (acc -> Name -> Name -> Graph -> acc) -> acc -> SCSet -> acc
111 | foldl f_acc acc
112 |     = foldlNames (\acc, f =>
113 |         foldlNames (\acc, g =>
114 |           foldl (\acc, c => f_acc acc f g c) acc)
115 |           acc)
116 |         acc
117 |
118 | insertGraph : (f, g : Name) -> Graph {- f g -} -> SCSet -> SCSet
119 | insertGraph f g ch s
120 |   = let s_f = lookupMap f s in
121 |     let s_fg = lookupSet g s_f in
122 |     insert f (insert g (insert ch s_fg) s_f) s

124 | ------------------------------------------------------------------------
125 | -- Actual size-change computations
126 |
127 | ||| Diagrammatic composition:
128 | ||| Given a (Change f g) and a (Change g h), compute a (Change f h)
129 | composeChange : Change {- f g -} -> Change {- g h -} -> Change {- f h -}
130 | composeChange
131 |     -- We use the SizeChange monoid: Unknown is a 0, Same is a neutral
132 |     = mult
133 |
134 | ||| Diagrammatic composition:
135 | ||| Given an (Graph f g) and an (Graph g h), compute an (Graph f h)
136 | composeGraph : Graph {- f g -} -> Graph {- g h -} -> Graph {- f h -}
137 | composeGraph a1 a2
138 |     = MkGraph
139 |         (composeChange a1.change a2.change)
140 |         (a1.seq ++ a2.seq)
141 |
142 | ||| Precompose a give an sc-graph & insert it in the worklist (unless it's already known)
143 | preCompose : (f : Name) -> Graph {- f g -} -> -- /!\ g bound later
144 |              SCSet ->
145 |              WorkList ->
146 |              (g : Name) -> (h : Name) -> Graph {- g h -} ->
147 |              WorkList
148 | preCompose f ch1 s work _ h ch2
149 |    = let ch : Graph {- f h -} = composeGraph ch1 ch2 in
150 |      if contains ch (lookupGraphs f h s) then
151 |        work
152 |      else
153 |        insert (f, h, ch) work
154 |
155 | ||| Precompose a given Arg change & insert it in the worklist (unless it's already known)
156 | postCompose : (h : Name) -> Graph {- g h -} -> -- /!\ g bound later
157 |               SCSet ->
158 |               WorkList ->
159 |               (f : Name) -> (g : Name) -> Graph {- f g -} ->
160 |               WorkList
161 | postCompose h ch2 s work f _ ch1
162 |    = let ch : Graph {- f h -} = composeGraph ch1 ch2 in
163 |      if contains ch (lookupGraphs f h s) then
164 |        work
165 |      else
166 |        insert (f, h, ch) work
167 |
168 | mutual
169 |   addGraph : (f, g : Name) -> Graph {- f g -} ->
170 |              WorkList ->
171 |              SCSet ->
172 |              SCSet
173 |   addGraph f g ch work s_in
174 |       = let s = insertGraph f g ch s_in
175 |             -- Now that (ch : Graph f g) has been added, we need to update
176 |             -- the graph with the paths it has extended i.e.
177 |
178 |             -- the ones start in g
179 |             after = selectDom g s
180 |             work_pre = foldl (preCompose f ch s) work after
181 |
182 |             -- and the ones ending in f
183 |             before = selectCod f s
184 |             work_post = foldl (postCompose g ch s) work_pre before in
185 |
186 |         -- And then we need to close over all of these new paths too
187 |         transitiveClosure work_post s
188 |
189 |   transitiveClosure : WorkList ->
190 |                       SCSet ->
191 |                       SCSet
192 |   transitiveClosure work s
193 |       = case pop work of
194 |              Nothing => s
195 |              Just ((f, g, ch), work) =>
196 |                addGraph f g ch work s
197 |
198 | -- find (potential) chain of calls to given function (inclusive)
199 | prefixCallSeq : NameMap (FC, Name) -> (g : Name) -> {- Exists \ f => -} CallSeq {- f g -}
200 | prefixCallSeq pred g = go g []
201 |   where
202 |     go : Name -> CallSeq -> CallSeq
203 |     go g seq
204 |         = let Just (l, f) = lookup g pred
205 |              | Nothing => seq in
206 |           if f == g then -- we've reached the entry point!
207 |             seq
208 |           else
209 |             go f ((l, g) :: seq)
210 |
211 | -- returns `Just a` iff there is no self-decreasing arc,
212 | -- i.e. there is no argument `n` with `a.change_{n,n} === Smaller`
213 | checkNonDesc : (a : Graph) -> Maybe Graph
214 | checkNonDesc a = if any selfDecArc a.change then Nothing else Just a
215 |   where
216 |     selfDecArc : (Nat, Vector1 SizeChange) -> Bool
217 |     selfDecArc (i, xs) = lookupOrd i xs == Just Smaller
218 |
219 | ||| Finding non-terminating loops
220 | findLoops : SCSet ->
221 |             NameMap {- $ \ f => -} (Maybe (Graph {- f f -} ))
222 | findLoops s
223 |     = -- An `SCSet` is non-terminating iff it contains a size graph that is
224 |       -- idempotent, i.e. stable under self-composition,
225 |       let loops = filterEndos (\a => composeChange a.change a.change == a.change) s in
226 |       -- and has no self-decreasing arcs.
227 |       map (foldMap checkNonDesc) loops
228 |     where
229 |       filterEndos : (Graph -> Bool) -> SCSet -> NameMap (List Graph)
230 |       filterEndos p = mapWithKey (\f, m => filter p (Prelude.toList (lookupSet f m)))
231 |
232 | findNonTerminatingLoop : SCSet -> Maybe (Name, Graph)
233 | findNonTerminatingLoop s = findNonTerminating (findLoops s)
234 |     where
235 |       -- select first non-terminating loop
236 |       -- TODO: find smallest
237 |       findNonTerminating : NameMap (Maybe Graph) -> Maybe (Name, Graph)
238 |       findNonTerminating = foldlNames (\acc, g, m => map (g,) m <+> acc) empty
239 |
240 | ||| Steps in a call sequence leading to a loop are also problematic
241 | setPrefixTerminating : {auto c : Ref Ctxt Defs} ->
242 |                        CallSeq -> Name -> Core ()
243 | setPrefixTerminating [] g = pure ()
244 | setPrefixTerminating (_ :: []) g = pure ()
245 | setPrefixTerminating ((l, f) :: p) g
246 |     = do setTerminating l f (NotTerminating (BadPath p g))
247 |          setPrefixTerminating p g

249 | addFunctions : {auto c : Ref Ctxt Defs} ->
250 |               Defs ->
251 |               List GlobalDef -> -- functions we're adding
252 |               NameMap (FC, Name) -> -- functions we've already seen (and their predecessor)
253 |               WorkList ->
254 |               Core (Either Terminating (WorkList, NameMap (FC, Name)))
255 | addFunctions defs [] pred work
256 |     = pure $ Right (work, pred)
257 | addFunctions defs (d1 :: ds) pred work
258 |     = do calls <- foldlC resolveCall [] d1.sizeChange
259 |          let Nothing = find isNonTerminating calls
260 |             | Just (d2, l, _) =>
261 |               do let g = d2.fullname
262 |                  let init = prefixCallSeq pred d1.fullname ++ [(l, g)]
263 |                  setPrefixTerminating init g
264 |                  pure $ Left (NotTerminating (BadPath init g))
265 |          let (ds, pred, work) = foldl addCall (ds, pred, work) (filter isUnchecked calls)
266 |          addFunctions defs ds pred work
267 |   where
268 |     resolveCall : List (GlobalDef, FC, (Name, Name, Graph)) ->
269 |                   SCCall ->
270 |                   Core (List (GlobalDef, FC, (Name, Name, Graph)))
271 |     resolveCall calls (MkSCCall g ch l)
272 |         = do Just d2 <- lookupCtxtExact g (gamma defs)
273 |                 | Nothing => do logC "totality.termination.calc" 7 $ do pure "Not found: \{show g}"
274 |                                 pure calls
275 |              pure ((d2, l, (d1.fullname, d2.fullname, MkGraph ch [(l, d2.fullname)])) :: calls)
276 |

279 |                   (List GlobalDef, NameMap (FC, Name), WorkList)
280 |     addCall (ds, pred, work_in) (d2, l, c)
281 |         = let work = insert c work_in in
282 |           case lookup d2.fullname pred of
283 |                Just _ =>
284 |                  (ds, pred, work)

286 |                  (d2 :: ds, insert d2.fullname (l, d1.fullname) pred, work)

307 | calcTerminating : {auto c : Ref Ctxt Defs} ->
308 |                   FC -> Name -> Core Terminating
309 | calcTerminating loc n
310 |     = do defs <- get Ctxt
311 |          logC "totality.termination.calc" 7 $ do pure $ "Calculating termination: " ++ show !(toFullNames n)
312 |          Just def <- lookupCtxtExact n (gamma defs)
313 |            | Nothing => undefinedName loc n
314 |          IsTerminating <- totRefs defs (nub !(addCases defs (keys (refersTo def))))
315 |            | bad => pure bad
316 |          Right (work, pred) <- initWork defs def
317 |            | Left bad => pure bad
318 |          let s = transitiveClosure work initSCSet
319 |          let Nothing = findNonTerminatingLoop s
320 |            | Just (g, loop) =>
321 |                ifThenElse (def.fullname == g)
322 |                  (pure $ NotTerminating (RecPath loop.seq))
323 |                  (do setTerminating EmptyFC g (NotTerminating (RecPath loop.seq))
324 |                      let init = prefixCallSeq pred g
325 |                      setPrefixTerminating init g
326 |                      pure $ NotTerminating (BadPath init g))
327 |          pure IsTerminating
328 |   where
329 |     addCases' : Defs -> NameMap () -> List Name -> Core (List Name)
330 |     addCases' defs all [] = pure (keys all)
331 |     addCases' defs all (n :: ns)
332 |         = case lookup n all of
333 |              Just _ => addCases' defs all ns
334 |              Nothing =>
335 |                if caseFn !(getFullName n)
336 |                   then case !(lookupCtxtExact n (gamma defs)) of
337 |                             Just def => addCases' defs (insert n () all)
338 |                                                   (keys (refersTo def) ++ ns)
339 |                             Nothing => addCases' defs (insert n () all) ns
340 |                   else addCases' defs (insert n () all) ns
