34 |   introLam : Name -> RigCount -> Term lhsCtxt -> Core IRawImp
35 |   introLam x rig ty = do
36 |     ty <- unelab env ty
37 |     defs <- get Ctxt
38 |     new_hole <- uniqueHoleName defs [] (nameRoot hole)
39 |     let iintrod = ILam replFC rig Explicit (Just x) ty (IHole replFC new_hole)
40 |     pure iintrod

42 |   introCon : Name -> Term lhsCtxt -> Core (List IRawImp)
43 |   introCon n ty = do
44 |     defs <- get Ctxt
45 |     ust <- get UST
46 |     Just gdef <- lookupCtxtExact n (gamma defs)
47 |       | _ => pure []
48 |     -- for now we only handle types with a unique constructor
49 |     let TCon _ _ _ _ _ _ cs _ = definition gdef
50 |       | _ => pure []
51 |     let gty = gnf env ty
52 |     ics <- for cs $ \ cons => do
53 |       Just gdef <- lookupCtxtExact cons (gamma defs)
54 |         | _ => pure Nothing
55 |       let nargs = lengthExplicitPi $ fst $ snd $ underPis (-1) [] (type gdef)
56 |       new_hole_names <- uniqueHoleNames defs nargs (nameRoot hole)
57 |       let new_holes = PHole replFC True <$> new_hole_names
58 |       let pcons = papply replFC (PRef replFC cons) new_holes
59 |       res <- catch
60 |         (do -- We're desugaring it to the corresponding TTImp
61 |             icons <- desugar AnyExpr lhsCtxt pcons
62 |             ccons <- checkTerm hidx {-is this correct?-} InExpr [] (MkNested []) env icons gty
63 |             newdefs <- get Ctxt
64 |             ncons <- normaliseHoles newdefs env ccons
65 |             icons <- unelab env ncons
66 |             pure (Just icons))
67 |         (\ _ => pure Nothing)
68 |       put Ctxt defs -- reset the context, we don't want any updates
69 |       put UST ust
70 |       pure res
71 |     pure (catMaybes ics)
