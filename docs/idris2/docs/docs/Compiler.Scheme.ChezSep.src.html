147 | ||| Compile a TT expression to a bunch of Chez Scheme files
148 | compileToSS : Ref Ctxt Defs -> String -> String -> ClosedTerm -> Core (Bool, List ChezLib)
149 | compileToSS c chez appdir tm = do
150 |   -- process native libraries
151 |   ds <- getDirectives Chez
152 |   libs <- findLibs ds
153 |   traverse_ copyLib libs
154 |   version <- coreLift $ chezVersion chez
155 |
156 |   -- get the material for compilation
157 |   cdata <- getCompileData False Cases tm
158 |   let ctm = forget (mainExpr cdata)
159 |   let ndefs = namedDefs cdata
160 |   let cui = getCompilationUnits ndefs
161 |
162 |   -- copy the support library
163 |   support <- readDataFile "chez/support-sep.ss"
164 |   let supportHash = show $ hash support
165 |   supportChanged <-
166 |     coreLift (readFile (appdir </> "support.hash")) >>= \case
167 |       Left err => pure True
168 |       Right fileHash => pure (fileHash /= supportHash)
169 |   when supportChanged $ do
170 |     Core.writeFile (appdir </> "support.ss") support
171 |     Core.writeFile (appdir </> "support.hash") supportHash
172 |
173 |   -- TODO: add extraRuntime
174 |   -- the problem with this is that it's unclear what to put in the (export) clause of the library
175 |   -- extraRuntime <- getExtraRuntime ds
176 |
177 |   -- for each compilation unit, generate code
178 |   chezLibs <- for cui.compilationUnits $ \cu => do
179 |     let chezLib = chezLibraryName cu
180 |
181 |     -- check if the hash has changed
182 |     -- TODO: also check that the .so file exists
183 |     let cuHash = show (hash cu)
184 |     hashChanged <-
185 |       coreLift (readFile (appdir </> chezLib <.> "hash")) >>= \case
186 |         Left err       => pure True
187 |         Right fileHash => pure (fileHash /= cuHash)
188 |
189 |     -- generate code only when necessary
190 |     when hashChanged $ do
191 |       defs <- get Ctxt
192 |       l <- newRef {t = List String} Loaded ["libc", "libc 6"]
193 |       s <- newRef {t = List String} Structs []
194 |
195 |       -- create imports + exports + header + footer
196 |       let imports = unwords
197 |             [ "(" ++
198 |                 maybe
199 |                   "unqualified"
200 |                   chezLibraryName
201 |                   (SortedMap.lookup cuid cui.byId)
202 |               ++ ")"
203 |             | cuid <- Prelude.toList cu.dependencies
204 |             ]
205 |       let exports = sepBy " " $ catMaybes
206 |             -- constructors don't generate Scheme definitions
207 |             [ case d of
208 |                 MkNmCon _ _ _ => Nothing
209 |                 _ => Just $ schName dn
210 |             | (dn, fc, d) <- cu.definitions
211 |             ]
212 |       let header =
213 |             "(library (" ++ fromString chezLib ++ ")\n"
214 |             ++ "  (export " ++ exports ++ ")\n"
215 |             ++ "  (import (chezscheme) (support) " ++ fromString imports ++ ")\n\n"
216 |       let footer = ")"
217 |
218 |       fgndefs <- traverse (Chez.getFgnCall version) cu.definitions
219 |       compdefs <- traverse (getScheme empty (Chez.chezExtPrim empty defaultLaziness) Chez.chezString defaultLaziness) cu.definitions
220 |       loadlibs <- traverse (map fromString . loadLib appdir) (mapMaybe fst fgndefs)
221 |
222 |       -- write the files
223 |       log "compiler.scheme.chez" 3 $ "Generating code for " ++ chezLib
224 |       Core.writeFile (appdir </> chezLib <.> "ss") $ build $ concat $
225 |         [header]
226 |         ++ map snd fgndefs  -- definitions using foreign libs
227 |         ++ compdefs
228 |         ++ loadlibs  -- foreign library load statements
229 |         ++ [footer]
230 |
231 |       Core.writeFile (appdir </> chezLib <.> "hash") cuHash
232 |
233 |     pure (MkChezLib chezLib hashChanged)
234 |
235 |   -- main module
236 |   main <- schExp empty (Chez.chezExtPrim empty defaultLaziness) Chez.chezString defaultLaziness 0 ctm
237 |   Core.writeFile (appdir </> "mainprog.ss") $ build $ sepBy "\n"
238 |     [ schHeader (map snd libs) [lib.name | lib <- chezLibs]
239 |     , collectRequestHandler
240 |     , main
241 |     , schFooter
242 |     ]
243 |
244 |   pure (supportChanged, chezLibs)

257 | ||| Chez Scheme implementation of the `compileExpr` interface.
258 | compileExpr :
259 |   Bool ->
260 |   Ref Ctxt Defs ->
261 |   Ref Syn SyntaxInfo ->
262 |   (tmpDir : String) -> (outputDir : String) ->
263 |   ClosedTerm -> (outfile : String) -> Core (Maybe String)
264 | compileExpr makeitso c s tmpDir outputDir tm outfile = do
265 |   -- set up paths
266 |   Just cwd <- coreLift currentDir
267 |        | Nothing => throw (InternalError "Can't get current directory")
268 |   let appDirSh  = outfile ++ "_app"  -- relative to the launcher shell script
269 |   let appDirRel = outputDir </> appDirSh  -- relative to CWD
270 |   let appDirAbs = cwd </> appDirRel
271 |   coreLift_ $ mkdirAll appDirRel
272 |
273 |   -- generate the code
274 |   chez <- coreLift $ findChez
275 |   (supportChanged, chezLibs) <- compileToSS c chez appDirRel tm
276 |
277 |   -- compile the code
278 |   logTime 2 "Make SO" $ when makeitso $ do
279 |     -- compile the support code
280 |     when supportChanged $ do
281 |       log "compiler.scheme.chez" 3 $ "Compiling support"
282 |       compileChezLibrary chez appDirRel (appDirRel </> "support.ss")
283 |
284 |     -- compile every compilation unit
285 |     compileChezLibraries chez appDirRel
286 |       [appDirRel </> lib.name <.> "ss" | lib <- chezLibs, lib.isOutdated]
287 |
288 |     -- touch them in the right order to make the timestamps right
289 |     -- even for the libraries that were not recompiled
290 |     for_ chezLibs $ \lib => do
291 |       log "compiler.scheme.chez" 3 $ "Touching " ++ lib.name
292 |       touch (appDirRel </> lib.name <.> "so")
293 |
294 |     -- compile the main program
295 |     compileChezProgram chez appDirRel (appDirRel </> "mainprog.ss")
296 |
297 |   -- generate the launch script
298 |   let outShRel = outputDir </> outfile
299 |   let launchTargetSh = appDirSh </> "mainprog" <.> (if makeitso then "so" else "ss")
300 |   if isWindows
301 |      then makeShWindows chez outShRel appDirSh launchTargetSh
302 |      else makeSh        chez outShRel appDirSh launchTargetSh
303 |   coreLift_ $ chmodRaw outShRel 0o755
304 |   pure (Just outShRel)
