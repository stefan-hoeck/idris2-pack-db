 8 | ||| a closing brace and call it a valid comment.

13 |   <+> many (is '-') <+> reject (is '}')  -- not a closing delimiter

16 | -- Identifier Lexer
17 | -- There are multiple variants.

19 | public export
20 | data Flavour = AllowDashes | Capitalised | Normal
21 |
22 | isIdentStart : Flavour -> Char -> Bool
23 | isIdentStart _           '_' = True
24 | isIdentStart Capitalised  x  = isUpper x || x > chr 160
25 | isIdentStart _            x  = isAlpha x || x > chr 160
26 |
27 | isIdentTrailing : Flavour -> Char -> Bool
28 | isIdentTrailing AllowDashes '-'  = True
29 | isIdentTrailing _           '\'' = True
30 | isIdentTrailing _           '_'  = True
31 | isIdentTrailing _            x   = isAlphaNum x || x > chr 160
32 |
33 | export %inline
34 | isIdent : Flavour -> String -> Bool
35 | isIdent flavour string =
36 |   case unpack string of
37 |     []      => False
38 |     (x::xs) => isIdentStart flavour x && all (isIdentTrailing flavour) xs
39 |
40 | export %inline
41 | ident : Flavour -> Lexer
42 | ident flavour =
43 |   (pred $ isIdentStart flavour) <+>
44 |     (many . pred $ isIdentTrailing flavour)

47 | isIdentNormal : String -> Bool
48 | isIdentNormal = isIdent Normal
