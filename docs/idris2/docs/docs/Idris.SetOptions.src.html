 52 | -- dissects a directory name, trying to extract
 53 | -- the corresponding package name and -version
 54 | pkgDir : (dirName : String) -> (ttcDirs : List Int) -> PkgDir
 55 | pkgDir dirName ttcDirs =
 56 |    -- Split the dir name into parts concatenated by "-"
 57 |    -- treating the last part as the version number
 58 |    -- and the initial parts as the package name.
 59 |    -- For reasons of backwards compatibility, we also
 60 |    -- accept hyphenated directory names without a part
 61 |    -- corresponding to a version number.
 62 |    case Lib.unsnoc $ split (== '-') dirName of
 63 |      (Nil, last) => MkPkgDir dirName last Nothing ttcDirs
 64 |      (init,last) =>
 65 |        case toVersion last of
 66 |          Just v  => MkPkgDir dirName (concat $ intersperse "-" init) (Just v) ttcDirs
 67 |          Nothing => MkPkgDir dirName dirName Nothing ttcDirs
 68 |   where
 69 |     toVersion : String -> Maybe PkgVersion
 70 |     toVersion = map MkPkgVersion
 71 |               . traverse parsePositive
 72 |               . split (== '.')
 73 |
 74 | listDirOrEmpty : String -> IO (List String)
 75 | listDirOrEmpty dir = either (const []) id <$> listDir dir

 77 | getPackageDirs : String -> IO (List PkgDir)
 78 | getPackageDirs dname = do
 79 |   packageDirNames <- listDirOrEmpty dname
 80 |   traverse (\d => pkgDir d <$> ttcVersions d) packageDirNames

 82 |     ttcVersions : String -> IO (List Int)
 83 |     ttcVersions dir = catMaybes . map parsePositive <$> listDirOrEmpty (dname </> dir)

 91 | candidateDirs :
 92 |     Ref Ctxt Defs =>
 93 |     String -> String -> PkgVersionBounds ->
 94 |     Core (List (String, Maybe PkgVersion))
 95 | candidateDirs dname pkgName bounds = do
 96 |   dirs <- coreLift (getPackageDirs dname)
 97 |   checkedDirs <- traverse check dirs

101 |     data CandidateError = OutOfBounds | TTCMismatch
102 |
103 |     checkNameAndBounds : PkgDir -> Either CandidateError PkgDir
104 |     checkNameAndBounds pkg =
105 |       if pkg.pkgName == pkgName && inBounds pkg.version bounds
106 |          then Right pkg
107 |          else Left OutOfBounds
108 |
109 |     checkTTCVersion : PkgDir -> Either CandidateError PkgDir
110 |     checkTTCVersion pkg =
111 |       if ttcVersion `elem` pkg.ttcVersions
112 |          then Right pkg
113 |          else Left TTCMismatch

115 |     unpack : PkgDir -> (String, Maybe PkgVersion)
116 |     unpack (MkPkgDir dirName pkgName ver _) =
117 |       ((dname </> dirName), ver)
118 |
119 |     check : PkgDir -> Core (Maybe (String, Maybe PkgVersion))
120 |     check pkg =
121 |       let checkedPkg = checkNameAndBounds pkg >>= checkTTCVersion
122 |       in
123 |       case checkedPkg of
124 |         Right pkg'       => pure . Just $ unpack pkg'
125 |         Left OutOfBounds => pure Nothing
126 |         Left TTCMismatch => do
