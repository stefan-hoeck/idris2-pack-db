110 | namespace Version
111 |   ||| Check if a Version is within the bounds of a PkgVersionBounds.
112 |   |||
113 |   ||| In addition to comparing major, minor, and patch version numbers,
114 |   ||| a Version with a tag is always considered larger than one with the
115 |   ||| same major, minor, and patch numbers but no tag.
116 |   |||
117 |   ||| Therefore, Version 0.1.0-abcd will be within the bounds >0.1.0 && <0.2.0.
118 |   ||| Similarly, Version 0.2.0-abcd will be _outside_ the bounds
119 |   ||| >0.1.0 && <=0.2.0.
120 |   export
121 |   inBounds : Version -> PkgVersionBounds -> Bool
122 |   inBounds (MkVersion (maj, min, patch) tag) bounds
123 |      = let v = MkPkgVersion (maj ::: [min, patch]) in
124 |        maybe True (\v' => if bounds.lowerInclusive
125 |                              then v >= v'
126 |                              else v > v' || (v == v' && tag /= Nothing)) bounds.lowerBound &&
127 |        maybe True (\v' => if bounds.upperInclusive
128 |                              then v < v' || (v == v' && tag == Nothing)
129 |                              else v < v') bounds.upperBound
130 |
131 |   -- "0.1.0-abcd" > "0.1.0"
132 |   inBoundsBecauseOfTag : inBounds (MkVersion (0,1,0) (Just "abcd"))
133 |                                   (MkPkgVersionBounds (Just $ MkPkgVersion (0 ::: [1,0])) False
134 |                                   (Just $ MkPkgVersion (0 ::: [2,0])) False) = True
135 |   inBoundsBecauseOfTag = Refl
136 |
137 |   -- not ("0.2.0-abcd" <= "0.2.0")
138 |   outOfBoundsBecauseOfTag : inBounds (MkVersion (0,2,0) (Just "abcd"))
139 |                                      (MkPkgVersionBounds (Just $ MkPkgVersion (0 ::: [1,0])) False
140 |                                      (Just $ MkPkgVersion (0 ::: [2,0])) True) = False
141 |   outOfBoundsBecauseOfTag = Refl
