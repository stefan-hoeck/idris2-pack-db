 49 | initIDESocketFile : String -> Int -> IO (Either String File)
 50 | initIDESocketFile h p = do
 51 |   osock <- socket AF_INET Stream 0
 52 |   case osock of
 53 |     Left fail => do
 54 |       putStrLn (show fail)
 55 |       putStrLn "Failed to open socket"
 56 |       exitWith (ExitFailure 1)
 57 |     Right sock => do
 58 |       res <- bind sock (Just (Hostname h)) p
 59 |       if res /= 0
 60 |         then pure (Left ("Failed to bind socket with error: " ++ show res))
 61 |         else
 62 |           do res <- listen sock
 63 |              if res /= 0
 64 |                 then
 65 |                   pure (Left ("Failed to listen on socket with error: " ++ show res))
 66 |                else
 67 |                  do p <- getSockPort sock
 68 |                     putStrLn (show p)
 69 |                     fflush stdout
 70 |                     res <- accept sock
 71 |                     case res of
 72 |                       Left err =>
 73 |                          pure (Left ("Failed to accept on socket with error: " ++ show err))
 74 |                       Right (s, _) =>
 75 |                          socketToFile s
