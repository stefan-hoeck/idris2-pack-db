 35 | mkOp : {auto s : Ref Syn SyntaxInfo} ->
 36 |        IPTerm -> Core IPTerm
 37 | mkOp tm@(PApp fc (PApp _ (PRef opFC kn) x) y)
 38 |   = do syn <- get Syn
 39 |        let raw = rawName kn
 40 |        let pop = if isOpName raw then OpSymbols else Backticked
 41 |        -- to check if the name is an operator we use the root name as a basic
 42 |        -- user name. This is because if the name is qualified with the namespace
 43 |        -- looking the fixity context will fail. A qualified operator would look
 44 |        -- like this: `M1.M2.(++)` which would not match its fixity namesapce
 45 |        -- that looks like this: `M1.M2.infixl.(++)`. However, since we only want
 46 |        -- to know if the name is an operator or not, it's enough to check
 47 |        -- that the fixity context contains the name `(++)`
 48 |        let rootName = UN (Basic (nameRoot raw))
 49 |        let asOp = POp fc (MkFCVal opFC
 50 |                 $ NoBinder (unbracketApp x)) (MkFCVal opFC (pop kn)) (unbracketApp y)
 51 |        if not (null (lookupName rootName (infixes syn)))
 52 |          then pure asOp
 53 |          else case dropNS raw of
 54 |            DN str _ => pure $ ifThenElse (isOpUserName (Basic str)) asOp tm
 55 |            _ => pure tm
