 67 | tokenise : Lexer ->
 68 |            Tokenizer a ->
 69 |            (line, col : Int) -> List (WithBounds a) ->
 70 |            List Char ->
 71 |            (List (WithBounds a), (StopReason, Int, Int, List Char))
 72 | tokenise reject tokenizer line col acc [] = (reverse acc, EndInput, (line, col, []))
 73 | tokenise reject tokenizer line col acc str
 74 |     = case scan reject [] str of
 75 |            Just _ => (reverse acc, (EndInput, line, col, str))
 76 |            Nothing => case getFirstMatch tokenizer str of
 77 |                            Right (toks, line', col', rest) =>
 78 |                                -- assert total because getFirstMatch must consume something
 79 |                                assert_total (tokenise reject tokenizer line' col' (toks ++ acc) rest)
 80 |                            Left reason => (reverse acc, reason, (line, col, str))
 81 |   where
 82 |     countNLs : List Char -> Nat
 83 |     countNLs str = List.length (filter (== '\n') str)
 84 |
 85 |     getCols : List Char -> Int -> Int
 86 |     getCols x c
 87 |         = case span (/= '\n') x of
 88 |                (incol, []) => c + cast (length incol)
 89 |                (incol, _) => cast (length incol)
 90 |
 91 |     -- get the next lexeme using the `Lexer` in argument, its position and the input
 92 |     -- Returns the new position, the lexeme parsed and the rest of the input
 93 |     -- If parsing the lexer fails, this returns `Nothing`
 94 |     getNext : (lexer : Lexer) ->  (line, col : Int) -> (input : List Char) -> Maybe (String, Int, Int, List Char)
 95 |     getNext lexer line col str =
 96 |       let Just (token, rest) = scan lexer [] str
 97 |             | _ => Nothing
 98 |           line' = line + cast (countNLs token)
 99 |           col' = getCols token col

105 |     getFirstMatch (Match lex fn) str

107 |                 | _ => Left NoRuleApply

109 |            in Right ([tok'], line', col', rest)

111 |         = let Just (beginTok', line', col' , rest) = getNext begin line col str
112 |                 | Nothing => Left NoRuleApply

115 |               end = endFn tag
