212 | ||| Names of all command line options (prefixed with "-" in case of
213 | ||| single-character option names and with "--" in case of multi-character
214 | ||| option names.
215 | export
216 | optionNames : List String
217 | optionNames = foldMap names descs
218 |
219 |   where
220 |     names : OptDescr a -> List String
221 |     names (MkOpt sns lns _ _) =
222 |       map (\c => "-\{String.singleton c}") sns ++ map ("--" ++) lns
223 |
224 | ||| Given the current directory (from which pack was invoked)
225 | ||| and an initial config assembled from the `pack.toml` files
226 | ||| in scope, generates the application
227 | ||| config and command to run from a list of command
228 | ||| line arguments.
229 | |||
230 | ||| @ c      : Type representing the command to run
231 | |||            We abstract over this type, because pack and
232 | |||            pack-admin accept different kinds of commands,
233 | |||            and both applications use this function to parse
234 | |||            the command line args
235 | |||
236 | ||| @ curDir : Current working directory
237 | ||| @ init   : Initial config (possibly assebled from `pack.toml` files)
238 | ||| @ args   : List of command line arguments
239 | export
240 | applyArgs :
241 |      (0 c : Type)
242 |   -> {auto _ : Command c}
243 |   -> (curDir : CurDir)
244 |   -> (init   : MetaConfig)
245 |   -> (args   : List String)
246 |   -> Either PackErr (MetaConfig, CommandWithArgs c)
247 | applyArgs c dir init args =
248 |   case getOpt RequireOrder descs args of
249 |     MkResult opts n  []      []       => do
250 |       cmd  <- readCommand c dir n
251 |       let lvl_m := lookup (cmdName $ fst cmd) init.levels
252 |           dflt  := defaultLevel $ fst cmd
253 |
254 |           init' = {logLevel := fromMaybe dflt lvl_m} init
255 |       conf <- foldlM (\c,f => f dir c) init' opts
256 |       Right (conf, cmd)
257 |
258 |     MkResult _    _ (u :: _) _        => Left (UnknownArg u)
259 |     MkResult _    _ _        (e :: _) => Left (ErroneousArg e)
260 |
