213 | solveDG :
214 |   (dg : DependencyGraph) ->
215 |   DependencyGraph
216 | solveDG dg = do
217 |   let cs = canSub dg
218 |   let False = null cs

257 | extractFVData :
258 |   Elaboration m =>
259 |   MonadError (Maybe UnificationError) m =>
260 |   (t : Type) ->
261 |   t ->
262 |   Vect l ArgDPair ->
263 |   Vect l String ->
264 |   m $ Vect l (Name, TTImp, Maybe TTImp)
265 | extractFVData t v ((Element fv isNamed) :: xs) (hn :: hns) = do
266 |   case t of
267 |     DPair myTy dNext => do
268 |       let (vv ** vRest) = v
269 |       quoteV <- quote vv
270 |       quoteT <- quote myTy
271 |       rest <- extractFVData (dNext vv) vRest xs hns
272 |       let retVal =
273 |         case quoteV of
274 |             IHole _ hh =>
275 |               if hh == hn then Nothing else Just quoteV
276 |             qv => Just qv
277 |       pure $ (argName fv, quoteT, retVal) :: rest
278 |     _ => do
279 |       qT <- quote t
280 |       throwError $ Just $ ExtractionError qT
281 | extractFVData t v [] [] = do
282 |   qT <- quote t
283 |   qV <- quote v
284 |   case t of
285 |     Equal x y =>
286 |       case qV of
287 |         INamedApp _ (INamedApp _ `(Builtin.Refl) _ _) _ _ => pure ()
288 |         _ => throwError $ Nothing
289 |     _ => throwError $ Just $ InternalError "DPairs don't correspond to each other. Should never occur."

292 | ||| Run unification
293 | unify' :
294 |   Elaboration m =>
295 |   MonadError (Maybe UnificationError) m =>
296 |   UnificationTask ->
297 |   m $ DependencyGraph
298 | unify' task = do
299 |   let 0 allFVsNamed = task.lhsAreNamed ++ task.rhsAreNamed
300 |   let allFreeVars = pushIn (task.lhsFreeVars ++ task.rhsFreeVars) allFVsNamed
301 |   let snocLFV : SnocVect task.lfv _ =
302 |     cast $ pushIn task.lhsFreeVars task.lhsAreNamed
303 |   let snocRFV : SnocVect task.rfv _ =
304 |     cast $ pushIn task.rhsFreeVars task.rhsAreNamed
305 |   (lhsNMap, lhsNames) <- genHoleNames snocLFV
306 |   (rhsNMap, rhsNames) <- genHoleNames snocRFV
307 |   let hole2N = IHole EmptyFC <$> mergeLeft lhsNMap rhsNMap
308 |   let allNames = lhsNames ++ rhsNames
309 |   -- Assemble the type, the value of which is all our free variables + proof of equality
310 |   let checkTargetType =
311 |     buildUpDPair snocLFV $
312 |       buildUpDPair snocRFV `(~(task.lhsExpr) ~=~ ~(task.rhsExpr))
313 |   -- Assemble the value (holes + Refl)
314 |   let checkTarget =
315 |     buildUpTarget (zip lhsNames snocLFV) $
316 |       buildUpTarget (zip rhsNames snocRFV) `(Refl)
317 |   logPoint DetailedDebug "unifyWithCompiler" [] "Target type: \{show checkTargetType}"
318 |   logPoint DetailedDebug "unifyWithCompiler" [] "Target value: \{show checkTarget}"
319 |   -- Instantiate target type
320 |   Just checkTargetType' : Maybe Type <-
321 |     try (Just <$> check checkTargetType) (pure Nothing)
322 |   | _ => throwError $ Just $ TargetTypeError checkTargetType
323 |   -- Run unification
324 |   Just checkTarget' : Maybe checkTargetType' <-
325 |     try (Just <$> check checkTarget) (pure Nothing)
326 |   | _ => throwError $ Just NoUnificationError
327 |   ctQuote <- quote checkTarget'
328 |   logPoint DetailedDebug "unifyWithCompiler" [] "Target value after quoting: \{show ctQuote}"
329 |   let vectNames = cast allNames
330 |   -- Extract unification results
331 |   uniResults <-
332 |     extractFVData checkTargetType' checkTarget' allFreeVars vectNames
333 |   logPoint DetailedDebug "unifyWithCompiler" [] "Raw unification results: \{show uniResults}"
334 |   -- Generate dependency graph
335 |   let allZipped = zip vectNames $ zip (task.lhsFreeVars ++ task.rhsFreeVars) uniResults
336 |   let dg = genDG $ makeFVData <$> allZipped
337 |   let dg = {fvData $= map {piInfo $= map $ substituteVariables hole2N}} dg
338 |   logPoint DetailedDebug "unifyWithCompiler" [] "Initial DG: \{show dg}"
339 |   let dg = subEmpties dg
340 |   logPoint DetailedDebug "unifyWithCompiler" [] "DG after subEmpties: \{show dg}"
341 |   let solved = solveDG dg
342 |   logPoint DetailedDebug "unifyWithCompiler" [] "Solved DG: \{show solved}"
343 |   pure solved
