216 | solveDG :
217 |   (dg : DependencyGraph) ->
218 |   DependencyGraph
219 | solveDG dg = do
220 |   let cs = canSub dg
221 |   let False = null cs
222 |   | _ => dg
223 |   let ds = doSub dg cs
224 |   -- DG <= DS because cs is non-empty, and every doSub may shrink the set of possibly substitutable variables

260 | extractFVData :
261 |   Elaboration m =>
262 |   MonadError UnificationError m =>
263 |   (t : Type) ->
264 |   t ->
265 |   Vect l ArgDPair ->
266 |   Vect l String ->
267 |   m $ Vect l (Name, TTImp, Maybe TTImp)
268 | extractFVData t v ((Element fv isNamed) :: xs) (hn :: hns) = do
269 |   case t of
270 |     DPair myTy dNext => do
271 |       let (vv ** vRest) = v
272 |       quoteV <- quote vv
273 |       quoteT <- quote myTy
274 |       rest <- extractFVData (dNext vv) vRest xs hns
275 |       let retVal =
276 |         case quoteV of
277 |             IHole _ hh =>
278 |               if hh == hn then Nothing else Just quoteV
279 |             qv => Just qv
280 |       pure $ (argName fv, quoteT, retVal) :: rest
281 |     _ => do
282 |       qT <- quote t
283 |       throwError $ ExtractionError $ qT
284 | extractFVData t v [] [] = do
285 |   qT <- quote t
286 |   qV <- quote v
287 |   case t of
288 |     Equal x y =>
289 |       case qV of
290 |         INamedApp _ (INamedApp _ `(Builtin.Refl) _ _) _ _ => pure ()
291 |         _ => throwError $ PostponeError

295 | ||| Run unification
296 | unify' :
297 |   Elaboration m =>
298 |   MonadError UnificationError m =>
299 |   UnificationTask ->
300 |   m $ DependencyGraph
301 | unify' task = do
302 |   let 0 allFVsNamed = task.lhsAreNamed ++ task.rhsAreNamed
303 |   let allFreeVars = pushIn (task.lhsFreeVars ++ task.rhsFreeVars) allFVsNamed
304 |   let snocLFV : SnocVect task.lfv _ =
305 |     cast $ pushIn task.lhsFreeVars task.lhsAreNamed
306 |   let snocRFV : SnocVect task.rfv _ =
307 |     cast $ pushIn task.rhsFreeVars task.rhsAreNamed
308 |   (lhsNMap, lhsNames) <- genHoleNames snocLFV
309 |   (rhsNMap, rhsNames) <- genHoleNames snocRFV
310 |   let hole2N = IHole EmptyFC <$> mergeLeft lhsNMap rhsNMap
311 |   let allNames = lhsNames ++ rhsNames
312 |   -- Assemble the type, the value of which is all our free variables + proof of equality
313 |   let checkTargetType =
314 |     buildUpDPair snocLFV $
315 |       buildUpDPair snocRFV `(~(task.lhsExpr) ~=~ ~(task.rhsExpr))
316 |   -- Assemble the value (holes + Refl)
317 |   let checkTarget =
318 |     buildUpTarget (zip lhsNames snocLFV) $
319 |       buildUpTarget (zip rhsNames snocRFV) `(Refl)
320 |   logPoint {level=DetailedTrace} "unifyWithCompiler" [] "Target type: \{show checkTargetType}"
321 |   logPoint {level=DetailedTrace} "unifyWithCompiler" [] "Target value: \{show checkTarget}"
322 |   -- Instantiate target type
323 |   Just checkTargetType' : Maybe Type <-
324 |     try (Just <$> check checkTargetType) (pure Nothing)
325 |   | _ => throwError $ TargetTypeError checkTargetType
326 |   -- Run unification
327 |   Just checkTarget' : Maybe checkTargetType' <-
328 |     try (Just <$> check checkTarget) (pure Nothing)
329 |   | _ => throwError NoUnificationError
330 |   ctQuote <- quote checkTarget'
331 |   logPoint {level=DetailedTrace} "unifyWithCompiler" [] "Target value after quoting: \{show ctQuote}"
332 |   let vectNames = cast allNames
333 |   -- Extract unification results
334 |   uniResults <-
335 |     extractFVData checkTargetType' checkTarget' allFreeVars vectNames
336 |   logPoint {level=DetailedTrace} "unifyWithCompiler" [] "Raw unification results: \{show uniResults}"
337 |   -- Generate dependency graph
338 |   let allZipped = zip vectNames $ zip (task.lhsFreeVars ++ task.rhsFreeVars) uniResults
339 |   let dg = genDG $ makeFVData <$> allZipped
340 |   let dg = {fvData $= map {piInfo $= map $ substituteVariables hole2N}} dg
341 |   -- logPoint {level=DetailedTrace} "unifyWithCompiler" [] "InitialDG: \{show dg}"
342 |   let dg = subEmpties dg
343 |   -- logPoint {level=DetailedTrace} "unifyWithCompiler" [] "DG after subEmpties: \{show dg}"
344 |   let solved = solveDG dg
345 |   -- logPoint {level=DetailedTrace} "unifyWithCompiler" [] "Solved DG: \{show solved}"
346 |   pure solved
