 51 | ||| Parse the start line of an HTTP request
 52 | ||| 
 53 | ||| Extracts HTTP method, target path, and HTTP version
 54 | ||| @ bs ByteString containing the start line
 55 | startLine : ByteString -> Either HTTPErr (Method,String,Version)
 56 | startLine bs =
 57 |   case toString <$> split SPACE (trim bs) of
 58 |     [m,t,v] => [| (\x,y,z => (x,y,z)) (method m) (pure t) (version v) |]
 59 |     _       => Left InvalidRequest

 62 | ||| Parse HTTP request headers
 63 | ||| 
 64 | ||| Recursively processes the header list and builds the header map
 65 | ||| @ hs Currently parsed header map
 66 | ||| @ t List of ByteStrings to parse
 67 | headers : Headers -> List ByteString -> Either HTTPErr Headers
 68 | headers hs []     = Right hs
 69 | headers hs (h::t) =
 70 |   case break (COLON ==) h of
 71 |     (xs,BS (S k) bv) =>
 72 |      let name := toLower (toString xs)
 73 |          val  := toString (trim $ tail bv)
 74 |       in headers (insert name val hs) t
 75 |     _                => Left InvalidRequest

 89 | ||| Assemble HTTP request object
 90 | ||| 
 91 | ||| Assembles parsed HTTP components into a complete Request object
 92 | ||| @ p Asynchronous pull stream of HTTP ByteStrings
 93 | export
 94 | assemble :
 95 |      HTTPPull (List ByteString) (HTTPStream ByteString)
 96 |   -> HTTPPull o (Maybe Request)
 97 | assemble p = Prelude.do
 98 |   Right (h,rem) <- C.uncons p | _ => pure Nothing
 99 |   (met,uri,vrs) <- injectEither (startLine h)
100 |   (hs,body)     <- foldPairE headers empty rem
101 |   let cl := contentLength hs
102 |       ct := contentType hs
103 |       queryParams := SortedMap.fromList $ []
104 |   when (cl > MaxContentSize) (throw ContentSizeExceeded)
105 |   pure $ Just (MkRequest met uri queryParams vrs hs cl ct $ C.take cl body)

137 | ||| serve is a function that handles a single client connection
138 | ||| 
139 | ||| Reads client request and returns response, then closes the connection
140 | ||| @ app Application handler function
141 | ||| @ cli Client socket
142 | covering
143 | serve : HTTPApplication -> Socket AF_INET -> Async Poll [] ()
144 | serve app cli =
145 |   flip guarantee (close' cli) $
146 |     mpull $ handleErrors (\(Here x) => stderrLn "\{x}") $
147 |          bytes cli 0xfff
148 |       |> request
149 |       |> handleRequest'
150 |   where
151 |     response : Maybe Request -> HTTPStream ByteString
152 |     response Nothing  = pure ()
153 |     response (Just r) =  app r |> mapOutput (encodeResponse' 200)

158 |         Left _   => emit badRequestHTTP |> writeTo cli
