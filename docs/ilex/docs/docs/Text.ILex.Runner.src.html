 96 | --------------------------------------------------------------------------------
 97 | -- Lexer run loop
 98 | --------------------------------------------------------------------------------
 99 |
100 | parameters {0 e,c,a  : Type}
101 |            {0 n      : Nat}
102 |            (todfa    : c -> DFA e c a)
103 |            (buf      : IBuffer n)
104 |
105 |   -- Reads next byte and determines the next automaton state
106 |   -- If this is the zero state, we backtrack to the latest
107 |   -- terminal state (in `last`), create a token from that
108 |   -- (in `app`) and start with a new token (in `loop`).
109 |   inner :
110 |        (dfa         : DFA e c a)             -- current finite automaton
111 |     -> (last        : Conv e c a)            -- last encountered terminal state
112 |     -> (start       : Nat)                   -- start of current token
113 |     -> (lastPos     : Nat)                   -- counter for last byte in `last`
114 |     -> {auto y      : Ix (S lastPos) n}      -- end position of `last` in the byte array
115 |     -> (vals        : SnocList $ Bounded a)  -- accumulated tokens
116 |     -> (pos         : Nat)                   -- reverse position in the byte array
117 |     -> {auto x      : Ix pos n}              -- position in the byte array
118 |     -> {auto 0 lte1 : LTE start (ixToNat y)}
119 |     -> {auto 0 lte2 : LTE start (ixToNat x)}
120 |     -> (cur         : Fin (S dfa.states))    -- current automaton state
121 |     -> Either (Bounded $ InnerError a e) (DFA e c a, SnocList (Bounded a))
122 |
123 |   -- Accumulates lexemes by applying the maximum munch strategy:
124 |   -- The largest matched lexeme is consumed and kept.
125 |   -- This consumes at least one byte for the next token and
126 |   -- immediately aborts with an error in case the current
127 |   -- byte leads to the zero state.
128 |   loop :
129 |        (dfa    : DFA e c a)            -- current finite automaton
130 |     -> (vals   : SnocList $ Bounded a) -- accumulated tokens
131 |     -> (pos    : Nat)                  -- reverse position in the byte array
132 |     -> {auto x : Ix pos n}             -- position in the byte array
133 |     -> Either (Bounded $ InnerError a e) (DFA e c a, SnocList (Bounded a))
134 |   loop dfa vals 0     = Right (dfa, vals)
135 |   loop dfa vals (S k) =
136 |     case (dfa.next `at` 0) `atByte` (buf `ix` k) of
137 |       0 => Left $ B (Byte $ buf `ix` k) (atPos $ ixToNat x)
138 |       s => inner dfa (dfa.term `at` s) (ixToNat x) k vals k s
139 |
140 |   -- Tries to create a new token from the current
141 |   -- state and append it to the list of tokens.
142 |   -- We need several pointers into the byte array:
143 |   --   `from` is the start index of the token
144 |   --   `ix`   is the end index of the token (`till` the corresponding reverse index)
145 |   --   `now` is the current position needed in case of
146 |   --   the encountered state being non-terminal
147 |   app :
148 |        SnocList (Bounded a)                  -- accumulated tokens
149 |     -> Conv e c a                            -- terminal state to use
150 |     -> Lazy (InnerError a e)                 -- error to use in case this is the `Bottom` state
151 |     -> (from        : Nat)                   -- start position of token
152 |     -> (till        : Nat)                   -- reverse end position + 1
153 |     -> (now         : Nat)                   -- current position
154 |     -> {auto ix     : Ix (S till) n}         -- end position
155 |     -> {auto 0  lte : LTE from (ixToNat ix)} -- current position
156 |     -> Either (Bounded $ InnerError a e) (DFA e c a, SnocList (Bounded a))
157 |   app sx c err from till now =
158 |     let bs := BS from (ixToNat ix)
159 |      in case c of
160 |           Bottom     => Left $ B err (atPos now)
161 |           Const cd v => loop (todfa cd) (sx :< B v bs) till
162 |           Ignore cd  => loop (todfa cd) sx till
163 |           Err   x => Left $ B (Custom x) bs
164 |           Txt   f => case f (toByteString buf from till) of
165 |             Left  x => Left $ B (Custom x) bs
166 |             Right (cd,v) => loop (todfa cd) (sx :< B v bs) till
167 |
168 |   inner dfa last start lastPos vals 0     cur =
169 |     app vals last EOI start lastPos (ixToNat x)
170 |   inner dfa last start lastPos vals (S k) cur =
171 |     let arr  := dfa.next `at` cur
172 |         byte := buf `ix` k
173 |      in case arr `atByte` byte of
174 |           FZ => app vals last (Byte byte) start lastPos (ixToNat x)
175 |           x  => case dfa.term `at` x of
176 |             Bottom => inner dfa last start lastPos vals k x
177 |             i      => inner dfa i    start k       vals k x
178 |
179 | lexFrom o (L ini todfa) pos buf =
180 |   case loop todfa buf (todfa ini) [<] pos of
181 |     Left (B x bs)  => Left $ PE o bs (fromIBuffer buf) x
182 |     Right (dfa,vs) => appEOI dfa o (fromIBuffer buf) pos vs

230 |     -> (col         : Nat)

233 |     -> Conv e c a
234 |     -> (from        : Nat)

236 |     -> {auto ix     : Ix (S till) n}
237 |     -> {auto 0  lte : LTE from (ixToNat ix)}

250 |
