111 | breakImpl : BreakInstruction -> (Bits8 -> Bool) -> ByteString -> BreakRes ByteString
112 | breakImpl x f bs =
113 |   case break f bs of
114 |     (pre, BS 0 _)      => Keep pre
115 |     (pre, pst@(BS (S k) bv)) => case x of
116 |       PostHit => Broken (nebs pre) (Just pst)
117 |       DropHit => Broken (nebs pre) (nebs $ BS k (tail bv))
118 |       TakeHit =>
119 |         let len := S pre.size
120 |          in Broken (Just $ take len bs) (nebs $ drop len bs)
