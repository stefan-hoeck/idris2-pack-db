 71 |   ||| An editable web form where the different fields of a
 72 |   ||| record can be edited and validated.
 73 |   |||
 74 |   ||| The record in question must be a *barbie*
 75 |   ||| (see the corresponding library), and this function takes as
 76 |   ||| input such a record of editors and returns an editor of
 77 |   ||| the record.
 78 |   |||
 79 |   ||| @ `rec`      : the barbie record type used to group the
 80 |   |||                field values.
 81 |   ||| @ `formNode` : used to group and display the different
 82 |   |||                form fields in a single HTML node.
 83 |   |||
 84 |   ||| @ `validate` : used send validation messages to a label or similar
 85 |   |||
 86 |   ||| Note: For uneditable record fields that should not appear in the
 87 |   |||       user interface, use a `dummy` editor.
 88 |   export
 89 |   form : rec (Editor . g) -> Editor (rec g)
 90 |   form edits =
 91 |        -- convert all value editors to record updaters
 92 |    let eflds   := bzipWith editField sings edits
 93 |
 94 |        -- initial value of the stream: all fields are missing
 95 |        -- (this will automatically be updated with the initial event of every
 96 |        -- input stream when `merge`ing them, so that non-mandatory field values
 97 |        -- will no longer be missing)
 98 |        missAll := the (rec (EditRes . g)) (bpure @{app} Missing)
 99 |     in E $ \recm => do

101 |          -- them to the optional initial record `recm`
