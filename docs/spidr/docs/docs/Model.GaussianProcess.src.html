 30 | ||| A Gaussian process is a collection of random variables, any finite number of which have joint
 31 | ||| Gaussian distribution. It can be viewed as a function from a feature space to a joint Gaussian
 32 | ||| distribution over a target space.
 33 | |||
 34 | ||| @features The shape of the feature domain.
 35 | public export
 36 | data GaussianProcess : (0 features : Shape) -> Type where
 37 |   ||| Construct a `GaussianProcess` as a pair of mean function and kernel.
 38 |   MkGP : MeanFunction features -> Kernel features -> GaussianProcess features
 39 |
 40 | posterior :
 41 |   GaussianProcess features ->
 42 |   Tensor [] F64 ->
 43 |   {s : _} -> (Tensor ((S s) :: features) F64, Tensor [S s] F64) ->
 44 |   Tag $ GaussianProcess features
 45 | posterior (MkGP priorMeanf priorKernel) noise (xTrain, yTrain) = do
 46 |   l <- tag $ cholesky $ !(priorKernel xTrain xTrain) + noise * identity
 47 |   let alpha = l.T \| (l |\ yTrain)
 48 |
 49 |       posteriorMeanf : MeanFunction features
 50 |       posteriorMeanf x = pure $ !(priorMeanf x) + !(priorKernel x xTrain) @@ alpha
 51 |
 52 |       posteriorKernel : Kernel features
 53 |       posteriorKernel x x' = pure
 54 |         $ !(priorKernel x x') - (l |\ !(priorKernel xTrain x)).T @@ (l |\ !(priorKernel xTrain x'))
 55 |
 56 |   pure $ MkGP posteriorMeanf posteriorKernel
 57 |
 58 | logMarginalLikelihood :
 59 |   GaussianProcess features ->
 60 |   Tensor [] F64 ->
 61 |   {s : _} -> (Tensor ((S s) :: features) F64, Tensor [S s] F64) ->
 62 |   Tag $ Tensor [] F64
 63 | logMarginalLikelihood (MkGP _ kernel) noise (x, y) = do
 64 |   l <- tag $ cholesky (!(kernel x x) + noise * identity)
 65 |   let alpha = l.T \| (l |\ y)
 66 |   pure $ - y @@ alpha / 2.0 - !(trace (log l)) - fromDouble (cast (S s)) * log (2.0 * pi) / 2.0

 68 | ||| A trainable model implementing vanilla Gaussian process regression. That is, regression with a
 69 | ||| Gaussian process as conjugate prior for homoscedastic Gaussian likelihoods. See the following
 70 | ||| for details:
 71 | |||
 72 | ||| Gaussian Processes for Machine Learning
 73 | ||| Carl Edward Rasmussen and Christopher K. I. Williams
 74 | ||| The MIT Press, 2006. ISBN 0-262-18253-X.
 75 | |||
 76 | ||| or
 77 | |||
 78 | ||| Pattern Recognition and Machine Learning, Christopher M. Bishop

 80 | data ConjugateGPRegression : (0 features : Shape) -> Type where
 81 |   ||| @gpFromHyperparameters Constructs a Gaussian process from the hyperparameters (presented as
 82 |   |||   a vector)
 83 |   ||| @hyperparameters The hyperparameters (excluding noise) presented as a vector.
 84 |   ||| @noise The likehood amplitude, or observation noise.
 85 |   MkConjugateGPR :
 86 |     {p : _} ->
 87 |     (gpFromHyperparameters : Tensor [p] F64 -> Tag $ GaussianProcess features) ->
 88 |     (hyperparameters : Tensor [p] F64) ->
 89 |     (noise : Tensor [] F64) ->
 90 |     ConjugateGPRegression features
 91 |
 92 | ||| A probabilistic model from feature values to a distribution over latent target values.
 93 | export
 94 | [Latent] ProbabilisticModel features [1] Gaussian (ConjugateGPRegression features) where
 95 |   marginalise (MkConjugateGPR mkGP gpParams _) x = do
 96 |     MkGP meanf kernel <- mkGP gpParams
 97 |     [| MkGaussian (expand 1 <$> meanf x) (expand 2 <$> kernel x x) |]
 98 |
 99 | ||| A probabilistic model from feature values to a distribution over observed target values.
100 | export
101 | [Observed] ProbabilisticModel features [1] Gaussian (ConjugateGPRegression features) where
102 |   marginalise gpr@(MkConjugateGPR _ _ noise) x = do
103 |     MkGaussian latentMean latentCov <- marginalise @{Latent} gpr x
104 |     pure $ MkGaussian latentMean $ latentCov + broadcast (expand 2 (noise * identity {n = S n}))

106 | ||| Fit the Gaussian process and noise to the specified data.
107 | export
108 | fit : (forall n . Tensor [n] F64 -> Optimizer $ Tensor [n] F64)
109 |   -> Dataset features [1]
110 |   -> ConjugateGPRegression features
111 |   -> Tag $ ConjugateGPRegression features
112 | fit optimizer (MkDataset x y) (MkConjugateGPR {p} mkPrior gpParams noise) = do
113 |   let objective : Tensor [S p] F64 -> Tag $ Tensor [] F64
114 |       objective params = do
115 |         let priorParams = slice [1.to (S p)] params
116 |         logMarginalLikelihood !(mkPrior priorParams) (slice [at 0] params) (x, squeeze y)
117 |
118 |   params <- optimizer (concat 0 (expand 0 noise) gpParams) objective
119 |
120 |   let mkPosterior : Tensor [p] F64 -> Tag $ GaussianProcess features
121 |       mkPosterior params' = posterior !(mkPrior params') (squeeze noise) (x, squeeze y)
122 |
123 |   pure $ MkConjugateGPR mkPosterior (slice [1.to (S p)] params) (slice [at 0] params)
124 |
