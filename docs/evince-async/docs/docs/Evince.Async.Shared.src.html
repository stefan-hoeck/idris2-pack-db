 34 | ||| Evaluate the top-level groups, running up to `cfg.jobs` of them
 35 | ||| concurrently. `jobs = 0` falls back to core's sequential walk; otherwise
 36 | ||| each batch of `jobs` groups is run together, with an abort
 37 | ||| check between batches so `--fail-fast` still short-circuits. A concurrent
 38 | ||| group's events are buffered and replayed under the lock when the group
 39 | ||| finishes, so groups' output never interleaves.
 40 | export
 41 | evalAsyncForest :
 42 |      Reporter (Async e [])
 43 |   -> Lock (Async e [])
 44 |   -> RunConfig
 45 |   -> AbortRef
 46 |   -> List (SpecTree (Async e []) ())
 47 |   -> Async e [] EvalResult
 48 | evalAsyncForest base lock cfg abortRef trees =
 49 |   case cfg.jobs of
 50 |     Z   => evalForest (lockedReporter base lock) cfg abortRef [] trees 0
 51 |     S k => go emptyResult (chunksOf (S k) trees)
 52 |   where
 53 |     runGroup : SpecTree (Async e []) () -> Async e [] EvalResult
 54 |     runGroup t = do
 55 |       buf <- liftIO (newIORef (the (SnocList Event) [<]))
 56 |       let buffered = MkReporter (\ev => liftIO (modifyIORef buf (:< ev)))
 57 |       r <- evalTree buffered cfg abortRef [] t 0
 58 |       evs <- liftIO (readIORef buf)
 59 |       lock.withLock (traverse_ base.onEvent (evs <>> []))
 60 |       pure r
 61 |
 62 |     go : EvalResult -> List (List (SpecTree (Async e []) ())) -> Async e [] EvalResult
 63 |     go acc []              = pure acc
 64 |     go acc (chunk :: rest) = do
 65 |       stop <- liftIO (readIORef abortRef)
 66 |       if stop
 67 |         then pure acc
 68 |         else do
 69 |           results <- parseq (map runGroup chunk)
 70 |           let merged = maybe emptyResult (foldl mergeResults emptyResult) results
 71 |           go (mergeResults acc merged) rest

 84 | runResultVia runLoop cfg spec = do
 85 |   out <- newIORef (the (Maybe EvalResult) Nothing)
