109 | fromDigs : Cast Nat b => Nat -> (b -> Maybe a) -> AutoTok e a
110 | fromDigs k f (x :: xs) =
111 |   if isDigit x then fromDigs (k * 10 + digit x) f xs else case f (cast k) of
112 |     Just v  => Succ v (x::xs)
113 |     Nothing => unknownRange p (x::xs)
114 | fromDigs k f [] = case f (cast k) of
115 |   Just v  => Succ v []
116 |   Nothing => unknownRange p []
117 |
118 | maybeFromDigs : Cast Nat b => a -> (b -> Maybe a) -> AutoTok e a
119 | maybeFromDigs v f (x::xs) =
120 |   if isDigit x then fromDigs (digit x) f xs else Succ v (x::xs)
121 | maybeFromDigs v f [] = Succ v []

123 | strictFromDigs : Cast Nat b => (b -> Maybe a) -> AutoTok e a
124 | strictFromDigs f (x::xs) =
