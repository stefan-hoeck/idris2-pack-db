 88 | ||| A magic interface for tuning the order of generation in derived generators
 89 | |||
 90 | ||| This interface defines a function `isConstructor` which can be implemented only by calling a macro `itIsConstructor`.
 91 | ||| If the argument of the implementation is not a name of a constructor, you will get a compilation error at this point.
 92 | |||
 93 | ||| `deriveFirst` function gets some information about given arguments in the current derivation task,
 94 | ||| and returns constructor's arguments that must be generated first in the derived generator, in order.
 95 | ||| Please notice that if the given order is incorrect, e.g. it suggests to generate some arguments which do not have any sufficient dependencies,
 96 | ||| this may lead to a compilation error of the derived generator.
 97 | ||| Any repetitions, mentions of already given arguments and mentions of arguments that will be generated by arguments earlier in the list
 98 | ||| will be simply ignored.
 99 | |||
100 | ||| Constructor arguments can be given by their number or their name (using name literals).
101 | |||
102 | ||| Typical usage is the following:
103 | |||
104 | ||| ```idris
105 | ||| data X : Nat -> Nat -> Type where
106 | |||   X1 : (n, m : Nat) -> n `LT` m => X n m
107 | |||   X2 : X n n
108 | |||
109 | ||| GenOrderTuning "X".dataCon where
110 | |||   isConstructor = itIsConstructor
111 | |||   deriveFirst gt gc = [`{m}, 2] -- 2 is the `LT` argument
112 | ||| ```

117 |                 (givenConArgs : List $ Fin isConstructor.fst.conInfo.args.length) ->
118 |                 List $ ConArg isConstructor.fst.conInfo

124 | -- adds base priorities of args which we depend on transitively
125 | refineBasePri : Num p => {con : _} -> FinMap con.args.length (Determination con, p) -> FinMap con.args.length (Determination con, p)
126 | refineBasePri ps = snd $ execState (SortedSet.empty {k=Fin con.args.length}, ps) $ traverse_ go $ List.allFins _ where
127 |   go : (visited : MonadState (SortedSet $ Fin con.args.length) m) =>
128 |        (pris : MonadState (FinMap con.args.length (Determination con, p)) m) =>
129 |        Monad m =>
130 |        Fin con.args.length ->
131 |        m ()
132 |   go curr = do
133 |
134 |     visited <- get
135 |     -- if we already managed the current item, then exit
136 |     let False = contains curr visited | True => pure ()
137 |     -- remember that we are in charge of the current item
138 |     let visited = insert curr visited
139 |     put visited
140 |
141 |     let Just (det, currPri) = lookup curr !(get @{pris}) | Nothing => pure ()
142 |
143 |     let unvisitedDeps = det.argsDependsOn `union` det.stronglyDeterminingArgs
144 |
145 |     -- run this for all dependences
146 |     for_ (unvisitedDeps `difference` visited) $ assert_total go
147 |
148 |     -- compute what needs to be added to the current priority
149 |     let addition = mapMaybe (map snd . lookup' !(get @{pris})) (Prelude.toList unvisitedDeps)
150 |     let newPri = foldl (+) currPri addition
151 |
152 |     -- update the priority of the currenly managed argument
153 |     modify $ updateExisting (mapSnd $ const newPri) curr
154 |
155 | propagateStrongDet, propagateDep : Ord a => FinMap con.args.length (Determination con, a) -> FinMap con.args.length (Determination con, a)
156 | -- propagate back along dependencies, but influence of this propagation should be approx. anti-propotrional to givens, hence `minus`
157 | propagateDep dets = dets <&> \(det, pri) => (det,) $ foldl (\x => maybe x (max x . snd) . lookup' dets) pri $ det.argsDependsOn
158 | -- propagate back along strong determinations
159 | propagateStrongDet dets =
160 |   foldl (\dets, (det, pri) => foldl (flip $ updateExisting $ map $ max pri) dets det.stronglyDeterminingArgs) dets dets
161 |
162 | propagatePri : Ord a => FinMap con.args.length (Determination con, a) -> FinMap con.args.length (Determination con, a)
163 | propagatePri dets = do
164 |   let next = propagatePriOnce dets
165 |   if ((==) `on` map snd) dets next
166 |     then dets
167 |     else assert_total propagatePri next
168 |   where
169 |     propagatePriOnce : FinMap con.args.length (Determination con, a) -> FinMap con.args.length (Determination con, a)
170 |     propagatePriOnce = propagateDep . propagateStrongDet

172 | data PriorityOrigin = Original | Propagated
173 |
174 | Eq PriorityOrigin where
175 |   Original   == Original   = True
176 |   Propagated == Propagated = True
177 |   _ == _ = False
178 |
179 | Ord PriorityOrigin where
180 |   compare Original   Original   = EQ
181 |   compare Original   Propagated = GT
182 |   compare Propagated Original   = LT
183 |   compare Propagated Propagated = EQ
184 |
185 | -- compute the priority
186 | -- priority is a count of given arguments, and it propagates back using `max` on strongly determining arguments and on arguments that depend on this
187 | -- additionally we take into account the number of outgoing strong determinations and count of dependent arguments
188 | assignPriorities : {con : _} -> FinMap con.args.length (Determination con) -> FinMap con.args.length (Determination con, Nat, PriorityOrigin, Nat)
189 | assignPriorities dets = do
190 |   let invStrongDetPwr = do
191 |     let _ : Monoid Nat = Additive
192 |     flip concatMap dets $ \det => fromList $ (,1) <$> det.stronglyDeterminingArgs.asList
193 |   -- the original priority is the count of already determined given arguments for each argument
194 |   let origPri = refineBasePri $ dets <&> \det => (det,) $ det.influencingArgs `minus` det.argsDependsOn.size
195 |   Fin.Map.mapWithKey' .| map snd origPri `zip` propagatePri origPri .| \idx, (origPri, det, newPri) =>
196 |     (det, newPri, if origPri == newPri then Original else Propagated, fromMaybe 0 (Fin.Map.lookup idx invStrongDetPwr) + det.argsDependsOn.size)
197 |
198 | findFirstMax : Ord p => List (a, b, p) -> Maybe (a, b)
199 | findFirstMax [] = Nothing
200 | findFirstMax ((x, y, pri)::xs) = Just $ go (x, y) pri xs where
201 |   go : (a, b) -> p -> List (a, b, p) -> (a, b)
202 |   go curr _       []                = curr
203 |   go curr currPri ((x, y, pri)::xs) = if pri > currPri then go (x, y) pri xs else go curr currPri xs
204 |
205 | searchOrder : {con : _} ->
206 |               (left : FinMap con.args.length $ Determination con) ->
207 |               List $ Fin con.args.length
208 | searchOrder left = do
209 |
210 |   -- find all arguments that are not stongly determined by anyone, among them find all that are not determined even weakly, if any
211 |   let notDetermined = filter (\(idx, det, _) => null det.stronglyDeterminingArgs) $ kvList $ assignPriorities left
212 |
213 |   -- choose the one from the variants
214 |   -- It's important to do so, since after discharging one of the selected variable, set of available variants can extend
215 |   -- (e.g. because of discharging of strong determination), and new alternative have more priority than current ones.
216 |   -- TODO to determine the best among current variants taking into account which indices are more complex (transitively!)
217 |   let Just (curr, currDet) = findFirstMax notDetermined
218 |     | Nothing => []
219 |
220 |   -- remove information about all currently chosen args
221 |   let next = removeDeeply .| Id curr .| removeDeeply currDet.argsDependsOn left
222 |
223 |   -- `next` is smaller than `left` because `curr` must be not empty
224 |   curr :: searchOrder (assert_smaller left next)

241 | --- Particular tactics ---
242 |
243 | ||| "Non-obligatory" means that some present external generator of some type
244 | ||| may be ignored even if its type is really used in a generated data constructor.
245 | namespace NonObligatoryExts
246 |
247 |   ||| Least-effort non-obligatory tactic is one which *does not use externals* during taking a decision on the order.
248 |   ||| It uses externals if decided order happens to be given by an external generator, but is not obliged to use any.
249 |   ||| It is seemingly most simple to implement, maybe the fastest and
250 |   ||| fits well when external generators are provided for non-dependent types.
251 |   export
252 |   [LeastEffort] ConstructorDerivator where
253 |     consGenExpr sig con givs fuel = do
254 |
255 |       -- Prepare local search context
256 |       let _ : NamesInfoInTypes = %search    -- I don't why it won't be found without this
257 |
258 |       -------------------------------------------------------------
259 |       -- Prepare intermediate data and functions using this data --
260 |       -------------------------------------------------------------
261 |
262 |       -- Compute left-to-right need of generation when there are non-trivial types at the left
263 |       argsTypeApps <- getTypeApps con
264 |
265 |       -- Decide how constructor arguments would be named during generation
266 |       let bindNames = bindNameRenamer . argName <$> fromList con.args
267 |       let argsTypeApps = do
268 |         let conArgNames : SortedSet Name = fromList $ argName <$> con.args
269 |         let intExprMapper : TTImp -> TTImp
270 |             intExprMapper $ IVar fc n = IVar fc $ if conArgNames `contains'` n then UN $ Basic $ bindNameRenamer n else n
271 |             intExprMapper x = x
272 |         argsTypeApps <&> {argApps $= map @{Compose} $ mapTTImp intExprMapper}
273 |
274 |       -- Get arguments which any other argument depends on
275 |       let dependees = dependees con.args
276 |
277 |       -- Form the expression of calling the current constructor
278 |       let callCons = do
279 |         let constructorCall = callCon con $ bindNames.withIdx <&> \(idx, n) => if contains idx dependees then implicitTrue else varStr n
280 |         let wrapImpls : Nat -> TTImp
281 |             wrapImpls Z     = constructorCall
282 |             wrapImpls (S n) = var `{Builtin.DPair.MkDPair} .$ implicitTrue .$ wrapImpls n
283 |         let consExpr = wrapImpls $ sig.targetType.args.length `minus` sig.givenParams.size
284 |         `(Prelude.pure {f=Test.DepTyCheck.Gen.Gen _} ~consExpr)
285 |
286 |       -- Derive constructor calling expression for given order of generation
287 |       let genForOrder : List (Fin con.args.length) -> m TTImp
288 |           -- ... state is the set of arguments that are already present (given or generated)
289 |           genForOrder order = map (foldr apply callCons) $ evalStateT givs $ for order $ \genedArg => do
290 |
291 |             -- Get info for the `genedArg`
292 |             let MkTypeApp typeOfGened argsOfTypeOfGened _ = index genedArg $ the (Vect _ $ TypeApp con) argsTypeApps
293 |
294 |             -- Acquire the set of arguments that are already present
295 |             presentArguments <- get
296 |
297 |             -- TODO to put the following check as up as possible as soon as it typecheks O_O
298 |             -- Check that those argument that we need to generate is not already present
299 |             let False = contains genedArg presentArguments
300 |               | True => pure id
301 |
302 |             -- Filter arguments classification according to the set of arguments that are left to be generated;
303 |             -- Those which are `Right` are given, those which are `Left` are needs to be generated.
304 |             let depArgs : Vect typeOfGened.args.length (Either (Fin con.args.length) TTImp) := argsOfTypeOfGened <&> \case
305 |               Right expr => Right expr
306 |               Left i     => if contains i presentArguments then Right $ varStr $ index i bindNames else Left i
307 |
308 |             -- Determine which arguments will be on the left of dpair in subgen call, in correct order
309 |             let subgeneratedArgs = mapMaybe getLeft $ toList depArgs
310 |
311 |             -- Make sure generated arguments will not be generated again
312 |             modify $ insert genedArg . union (fromList subgeneratedArgs)
313 |
314 |             -- Form a task for subgen
315 |             let (subgivensLength ** subgivens) = mapMaybe (\(ie, idx) => (idx,) <$> getRight ie) $ depArgs `zip` Fin.range
316 |             let subsig : GenSignature := MkGenSignature typeOfGened $ fromList $ fst <$> toList subgivens
317 |             let Yes Refl = decEq subsig.givenParams.size subgivensLength
318 |               | No _ => fail "INTERNAL ERROR: error in given params set length computation"

410 | ||| "Obligatory" means that is some external generator is present and a constructor has
411 | ||| an argument of a type which is generated by this external generator, it must be used
412 | ||| in the constructor's generator.
413 | |||
414 | ||| Dependent types are important here, i.e. generator ``(a : _) -> (b ** C a b)``
415 | ||| is considered to be a generator for the type ``C``.
416 | ||| The problem with obligatory generators is that some external generators may be incompatible.
417 | |||
418 | |||   E.g. once we have ``(a : _) -> (b ** C a b)`` and ``(a ** b ** C a b)`` at the same time,
419 | |||   once ``C`` is used in the same constructor, we cannot guarantee that we will use both external generators.
420 | |||
421 | |||   The same problem is present once we have external generators for ``(a : _) -> (b : T ** C a b)`` and ``(b : T ** D b)`` at the same time,
422 | |||   and both ``C`` and ``D`` are used in the same constructor with the same parameter of type ``T``,
423 | |||   i.e. when constructor have something like ``C a b -> D b -> ...``.
424 | |||
425 | |||   Notice, that this problem does not arise in constructors of type ``C a b1 -> D b2 -> ...``
426 | |||
427 | ||| In this case, we cannot decide in general which value of type ``T`` to be used for generation is we have to use both generators.
428 | ||| We can either fail to generate a value for such constructor (``FailFast`` tactic),
429 | ||| or alternatively we can try to match all the generated values of type ``T`` from both generators
430 | ||| using ``DecEq`` and leave only intersection (``DecEqConflicts`` tactic).
