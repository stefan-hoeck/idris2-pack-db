 17 | isSimpleBindVar : TTImp -> Bool
 18 | isSimpleBindVar $ IBindVar {} = True
 19 | isSimpleBindVar _             = False

 21 | export
 22 | canonicConsBody : DeriveBodyRhsForCon => DerivationClosure m => GenSignature -> Name -> Con -> m $ List Clause
 23 | canonicConsBody sig name con = do
 24 |
 25 |   -- Get file position of the constructor definition (for better error reporting)
 26 |   let conFC = getFC con.type
 27 |
 28 |   -- Normalise the types in constructor; expand functions that are used as types, if possible
 29 |   con <- normaliseCon con
 30 |
 31 |   -- Acquire constructor's return type arguments
 32 |   let (conRetTy, conRetTypeArgs) = unAppAny con.type
 33 |   let conRetTypeArgs = conRetTypeArgs <&> getExpr
 34 |
 35 |   -- Match lengths of `conRetTypeArgs` and `sig.targetType.args`
 36 |   let Yes conRetTypeArgsLengthCorrect = conRetTypeArgs.length `decEq` sig.targetType.args.length
 37 |     | No _ => failAt conFC "INTERNAL ERROR: length of the return type does not equal to the type's arguments count"
 38 |
 39 |   let conRetTypeArg : Fin sig.targetType.args.length -> TTImp
 40 |       conRetTypeArg idx = index' conRetTypeArgs $ rewrite conRetTypeArgsLengthCorrect in idx
 41 |
 42 |   -- Determine names of the arguments of the constructor (as a function)
 43 |   let conArgNames = fromList $ argName <$> con.args
 44 |
 45 |   -- For given arguments, determine whether they are
 46 |   --   - just a free name
 47 |   --   - repeated name of another given parameter (need of `decEq`)
 48 |   --   - (maybe, deeply) constructor call (need to match)
 49 |   --   - function call on a free param (need to use "inverted function" filtering trick)
 50 |   --   - something else (cannot manage yet, unless it is fully determined by other given arguments)
 51 |   let deepConsApps : Vect _ $ Either (String, TTImp, List Name) _ := sig.givenParams.asVect <&> \idx => do
 52 |     let argExpr = conRetTypeArg idx
 53 |     let (ei, fns) = runWriter $ runEitherT {e=String} {m=Writer _} $ analyseDeepConsApp True conArgNames argExpr
 54 |     flip mapFst ei $ \err =>
 55 |       ("Argument #\{show idx} of \{show con.name} with given type arguments [\{showGivens sig}] is not supported, " ++
 56 |        "argument expression: \{show argExpr}, reason: \{err}", argExpr, fns)
 57 |   let allAppliedFreeNames = foldMap (SortedSet.fromList . either (snd . snd) (map fst . fst)) deepConsApps
 58 |   let bindAppliedFreeNames : TTImp -> TTImp
 59 |       bindAppliedFreeNames orig@(IVar _ n) = if contains n allAppliedFreeNames then bindVar $ bindNameRenamer n else orig
 60 |       --                              /---------------------------------------------^^^^^^^
 61 |       -- `bindVar` instead of `var` here is because this expression can be earlier than the real binding, thus undeclared,
 62 |       -- and the compiler turns unnecessary `IBindVar`s into `IVar`s by itself
 63 |       bindAppliedFreeNames x = x
 64 |   deepConsApps <- for deepConsApps $ \case
 65 |     Right x => pure x
 66 |     Left (err, argExpr, fns) => if null $ (allVarNames' argExpr `intersection` conArgNames) `difference` allAppliedFreeNames
 67 |       then pure (filter (contains' conArgNames) fns <&> (, neutral) ** const $ mapTTImp bindAppliedFreeNames argExpr)
 68 |       else failAt conFC err
 69 |
 70 |   -- Acquire LHS bind expressions for the given parameters
 71 |   -- Determine pairs of names which should be `decEq`'ed
 72 |   let getAndInc : forall m. MonadState Nat m => m Nat
 73 |       getAndInc = get <* modify S
 74 |   ((givenConArgs, decEqedNames, _), bindExprs) <-
 75 |     runStateT (empty, empty, 0) {stateType=(SortedSet Name, SortedSet (String, String), Nat)} {m} $
 76 |       for deepConsApps $ \(appliedNames ** bindExprF) => do
 77 |         renamedAppliedNames <- for appliedNames.asVect $ \(name, typeDetermined) => do
 78 |           let bindName = bindNameRenamer name
 79 |           if cast typeDetermined
 80 |             then pure $ const implicitTrue -- no need to match type-determined parameter by hand
 81 |             else if contains name !get
 82 |             then do
 83 |               -- I'm using a name containing chars that cannot be present in the code parsed from the Idris frontend
 84 |               let substName = "to_be_deceqed^^" ++ bindName ++ show !getAndInc
 85 |               modify $ insert (bindName, substName)
 86 |               pure $ \alreadyMatchedRenames => bindVar $ if contains substName alreadyMatchedRenames then bindName else substName
 87 |             else modify (insert name) $> const (bindVar bindName)
 88 |         let _ : Vect appliedNames.length $ SortedSet String -> TTImp = renamedAppliedNames
 89 |         pure $ \alreadyMatchedRenames => bindExprF $ \idx => index idx renamedAppliedNames $ alreadyMatchedRenames
 90 |   let bindExprs = \alreadyMatchedRenames => bindExprs <&> \f => f alreadyMatchedRenames
 91 |
 92 |   -- Build a map from constructor's argument name to its index
 93 |   let conArgIdxs = SortedMap.fromList $ mapI con.args $ \idx, arg => (argName arg, idx)
 94 |
 95 |   -- Determine indices of constructor's arguments that are given
 96 |   givenConArgs <- for givenConArgs.asList $ \givenArgName => do
 97 |     let Just idx = lookup givenArgName conArgIdxs
 98 |       | Nothing => failAt conFC "INTERNAL ERROR: calculated given `\{show givenArgName}` is not found in an arguments list of the constructor"
 99 |     pure idx
100 |
101 |   -- Equalise index values which must be propositionally equal to some parameters
102 |   -- NOTE: Here I do all `decEq`s in a row and then match them all against `Yes`.
103 |   --       I could do this step by step and this could be more effective in large series.
104 |   let deceqise : (lhs : Vect sig.givenParams.asList.length TTImp -> TTImp) -> (rhs : TTImp) -> Clause
105 |       deceqise lhs rhs = step lhs empty $ orderLikeInCon decEqedNames where
106 |
107 |         step : (withlhs : Vect sig.givenParams.asList.length TTImp -> TTImp) ->
108 |                (alreadyMatchedRenames : SortedSet String) ->
109 |                (left : List (String, String)) ->
110 |                Clause
111 |         step withlhs matched [] = PatClause EmptyFC .| withlhs (bindExprs matched) .| rhs
112 |         step withlhs matched ((orig, renam)::rest) =
113 |           WithClause EmptyFC (withlhs $ bindExprs matched) MW
114 |             `(Decidable.Equality.decEq ~(varStr renam) ~(varStr orig))
115 |             Nothing []
116 |             [ -- happy case
117 |               step ((.$ `(Prelude.Yes Builtin.Refl)) . withlhs) (insert renam matched) rest
118 |             , -- empty case
119 |               PatClause EmptyFC .| withlhs (bindExprs matched) .$ `(Prelude.No _) .| `(empty)
120 |             ]
121 |
122 |         -- Order pairs by the first element like they are present in the constructor's signature
123 |         orderLikeInCon : Foldable f => f (String, String) -> List (String, String)
124 |         orderLikeInCon = do
125 |           let conArgStrNames = mapMaybe argStrName con.args
126 |           let conNameToIdx : SortedMap _ $ Fin conArgStrNames.length := fromList $ mapI conArgStrNames $ flip (,)
127 |           let [AsInCon] Ord (String, String) where
128 |                 compare (origL, renL) (origR, renR) = comparing (lookup' conNameToIdx) origL origR <+> compare renL renR
129 |           Prelude.toList . foldl insert' (empty @{AsInCon})
130 |           where
131 |             argStrName : Arg -> Maybe String
132 |             argStrName $ MkArg {name=Just $ UN (Basic n), _} = Just n
133 |             argStrName _                                     = Nothing
134 |
135 |   -- Form the declaration cases of a function generating values of particular constructor
136 |   let fuelArg = "^cons_fuel^" -- I'm using a name containing chars that cannot be present in the code parsed from the Idris frontend
137 |   pure $
138 |     -- Happy case, given arguments conform out constructor's GADT indices
139 |     [ deceqise (callCanonic sig name $ bindVar fuelArg) !(consGenExpr sig con .| fromList givenConArgs .| varStr fuelArg) ]
140 |     ++ if all isSimpleBindVar $ bindExprs SortedSet.empty then [] {- do not produce dead code if the happy case handles everything already -} else
141 |       -- The rest case, if given arguments do not conform to the current constructor then return empty generator
142 |       [ callCanonic sig name implicitTrue (replicate _ implicitTrue) .= `(empty) ]
