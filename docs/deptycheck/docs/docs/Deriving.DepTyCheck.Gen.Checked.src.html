 50 | public export
 51 | record ExternalGenSignature where
 52 |   constructor MkExternalGenSignature
 53 |   targetType : TypeInfo
 54 |   {auto 0 targetTypeCorrect : AllTyArgsNamed targetType}
 55 |   givenParams : SortedMap (Fin targetType.args.length) (ArgExplicitness, Name)
 56 |   givensOrder : Vect givenParams.size $ Fin givenParams.size -- must be a permutation
 57 |   {gendParamsCnt : _}
 58 |   gendOrder   : Vect gendParamsCnt $ Fin gendParamsCnt -- must be a permutation

 86 | callExternalGen sig topmost fuel values =
 87 |   foldl (flip apply) (appFuel topmost fuel) $ reorder sig.givensOrder (sig.givenParams.asVect `zip` values) <&> \case

111 |                         Maybe (Name, Subset ExternalGenSignature $ \extSig => extSig.givenParams.size = intSig.givenParams.size)
112 |   lookupLengthChecked intSig m = lookup intSig m >>= \(extSig, name) => (name,) <$>
113 |                                    case decEq extSig.givenParams.size intSig.givenParams.size of
114 |                                       Yes prf => Just $ Element extSig prf
