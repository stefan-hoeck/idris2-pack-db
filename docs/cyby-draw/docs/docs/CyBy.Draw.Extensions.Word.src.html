 77 | -- A quick note on the mechanism used in this module:
 78 | -- As Word is based on XML structure, its JavaScript API
 79 | -- allows it to be extracted as a string. Therefore,
 80 | -- information about the structure is saved inside certain
 81 | -- XML nodes.
 82 | -- To export a new structure from CyBy-Draw to Word, the
 83 | -- JS API offers a simple image insertion function that
 84 | -- generates the required nodes. To retrieve the structure
 85 | -- at a later time, the MOL graph is stored in a metadata
 86 | -- node in the SVG. However, if a structure (generated by
 87 | -- the CyBy-Draw add-in) is to be modified, the entire SVG
 88 | -- node must be replaced using the JS string function
 89 | -- replace. As the size of the new SVG may differ, the old
 90 | -- Word-generated values must be replaced. Currently, new
 91 | -- size values are generated by inserting a temporary image
 92 | -- and extracting the values from the XML. The temporary
 93 | -- image is then deleted.
 94 | -- To retrieve a structure from Word, the cursor selection
 95 | -- XML string can be searched for the first occurrence of
 96 | -- the SVG node and its first attribute, 'xmlns'. As Word
 97 | -- always adds a 'viewBox' attribute and places it as the
 98 | -- first attribute of an SVG node, only the current
 99 | -- selection of SVGs generated by CyBy-Draw are searched
100 | -- for. As the RegEx is non-greedy, only the first SVG
101 | -- occurrence is relevant.
102 | exportImgEmptSel : HasIO io => LogLevel => Selection -> (svg,mol : String) -> io ()
103 | exportImgEmptSel s svg mol = do
104 |   debug "Selection is empty or no svg is present for replacing the structure"
105 |   -- encode and insert the image to Word
106 |   -- the JS API creates an xml entry, where the graph is stored
107 |   -- in the svg reference for later use
108 |   ignore $ insertInlinePictureFromB64 s (btoa svg)
109 |   debug "exportImage succcesfull"
110 |
111 | exportImage : LogLevel => (svg,mol : String) -> Prog ()
112 | exportImage svg mol =
113 |   wordRun $ \c => do
114 |     debug "Begin of function `exportImage`"
115 |     -- replace the selected svg (or the first in the selection)
116 |     -- with the updated structure if an svg is selected
117 |     -- if the selection is empty or does not include an svg,
118 |     -- insert the new structure after the selection / cursor
119 |     s <- getSelection c
120 |     False <- isEmpty s | True => exportImgEmptSel s svg mol
121 |
122 |     -- load the whole selection as xml
123 |     ooxml <- getSelectionOoxml c s
124 |
125 |     let True    := hasCyBySvg ooxml | False => exportImgEmptSel s svg mol
126 |         idSel   := extractImageIdWordSel ooxml
127 |         svgTemp := btoa (createTempSvg svg)
128 |
129 |     -- TODO: This is a hack and should be removed.
130 |     -- creating a new InlinePicture for extracting the
131 |     -- word generated size, for that all image sizes are equally (generated)
132 |     -- (Word slightly adjusts sizes based on their position in the file,
133 |     -- therefore not all structures have the same length!)
134 |     -- maybe change to appending the image to the end of the file
135 |     tempImg <- insertInlinePictureFromB64 s svgTemp
136 |     load c tempImg "" -- TODO: what does this do?
137 |     syncContext c
138 |
139 |     tempOoxml <- getOoxml c
140 |
141 |     let tempId   := extractTempImageId tempOoxml
142 |
143 |     -- deleting the temp image, as the sizes are captured
144 |     deleteInlinePicture tempImg
145 |
146 |     case extractTempSvgSize tempOoxml tempId of
147 |       Nothing => do
148 |         debug "Failed to extract the size of the temporary image."
149 |         debug "Abort updating structure and cleanup."
150 |       Just p => do
151 |         trace "Id of the temp image: \{tempId}"
152 |         trace "Size of the temp image: \{show p}"
153 |
154 |         -- replace the first occurring svg with the updated one
155 |         -- and replace the new sizes
156 |         replaceOoxml s (replaceSvgAndSize ooxml svg idSel p)
157 |         debug "Function `exportImage` successful"
158 |
159 | exportImageToWord : LogLevel => (svg,molFile : String) -> JSIO ()
160 | exportImageToWord svg mol = runDeflt $ exportImage svg mol

163 | importImageFromWord f =
164 |   wordRun $ \c => do
165 |     debug "Begin of function `importImageFromWord`"
166 |     -- return an empty string if the selection is empty
167 |     s <- getSelection c
168 |     False <- isEmpty s | True => debug "Selection is empty" >> primIO (f "")
169 |
170 |     -- load the whole selection as xml
171 |     ooxml <- getSelectionOoxml c s
172 |
173 |     -- extracting the MOL file directly from the xml structure
174 |     -- of the current selection
175 |     -- if there are several cyby-draw generated structures, the
176 |     -- first in the selection is imported
177 |     case extractMetadata ooxml of
178 |       "" => debug "No MOL-File found"
179 |       g  => debug "Function `exportImage` succcesfull" >> primIO (f g)
