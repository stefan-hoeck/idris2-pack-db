414 |
415 | ||| Repeatedly consumes a strict prefix of a list of characters
416 | ||| until the whole list is consumed. Drops all white space
417 | ||| it encounters (unsing `Prelude.isSpace` to determine, what is
418 | ||| a whitespace character).
419 | |||
420 | ||| This assumes that every token is on a single line. Therefore, it is
421 | ||| more efficient than `multilineDropSpaces`, because it does not have
422 | ||| to traverse the consumed characters to find line breaks.
423 | |||
424 | ||| This is provably total, due to the strictness of the consuming
425 | ||| function.
426 | public export
427 | singleLineDropSpaces :
428 |      Tok True e a
429 |   -> String
430 |   -> Either (Bounded $ ParseError Void e) (List $ Bounded a)
431 | singleLineDropSpaces f s = go begin [<] (unpack s) suffixAcc
432 |
433 |   where
434 |     go : Position
435 |       -> SnocList (Bounded a)
436 |       -> (ts : List Char)
437 |       -> (0 acc : SuffixAcc ts)
438 |       -> Either (Bounded $ ParseError Void e) (List $ Bounded a)
439 |     go p1 sx []       _        = Right $ sx <>> []
440 |     go p1 sx ('\n'::cs) (SA r) = go (incLine p1) sx cs r
441 |     go p1 sx (c::cs)    (SA r) =
442 |       if isSpace c then go (incCol p1) sx cs r
443 |       else case f (c::cs) of
444 |         Succ v xs2 @{p}     =>
