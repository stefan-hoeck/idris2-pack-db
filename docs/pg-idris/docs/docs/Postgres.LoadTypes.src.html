103 | ||| Using the groupings of Postgres string names for types that will
104 | ||| map to each PType, parse the given string to a PType (or POther)
105 | parseType : String -> PType
106 | parseType type = case isElem True typeSearch of
107 |                       (No _)  => POther type
108 |                       (Yes e) => case elemToFin e of
109 |                                       0  => arrayOrNot (fst typeSpec) PInteger
110 |                                       1  => arrayOrNot (fst typeSpec) PDouble
111 |                                       2  => arrayOrNot (fst typeSpec) PChar
112 |                                       3  => arrayOrNot (fst typeSpec) PBoolean
113 |                                       4  => arrayOrNot (fst typeSpec) PDate
114 |                                       5  => arrayOrNot (fst typeSpec) PTime
115 |                                       6  => arrayOrNot (fst typeSpec) PDatetime
116 |                                       7  => arrayOrNot (fst typeSpec) PString
117 |                                       8  => arrayOrNot (fst typeSpec) PJson
118 |                                       9  => arrayOrNot (fst typeSpec) PUuid
119 |                                       10 => arrayOrNot (fst typeSpec) POid
120 |   where
121 |     ||| First element of tuple is true if the type
122 |     ||| is an array. Second element of tuple is the
123 |     ||| name of the non-array type (because array types
124 |     ||| are named the same as non-array types but with
125 |     ||| a leading underscore).

148 | ||| Load Postgres types into a type dictionary. This is needed so that future queries
149 | ||| can identify the types of columns in responses.

151 | pgLoadTypes : HasIO io => Conn -> io (Either String TypeDictionary)
152 | pgLoadTypes conn =
153 |   do Right (r ** 2 ** resultset) <- liftIO $ pgStringResultsQuery {types=empty} False typeQuery conn 
154 |        | Right (_ ** c ** _) => pure $ Left $ "ERROR: expected 2 columns but got " ++ (show c)
155 |        | Left err            => pure $ Left $ "ERROR: " ++ err
156 |      -- could change following to successfully parse a subset of types even when failing on one of them.
157 |      Right types <- pure $ traverse typeResult resultset
158 |        | Left err => pure $ Left $ "ERROR: " ++ err
159 |      pure $ Right $ typeDictionary $ toList types
