115 | parseURL : String -> Either String (URL String System.Path.Path (List (String, String)))
116 | parseURL url = do
117 |   url <- mapFst printErr (Simple.parse url)
118 |   let pp = url.path
119 |   let qry = parseSearch url.search
120 |   path <- maybeToEither "could not parse path \{pp}" (Path.tryParse pp)
121 |   pure ({path := path, search := qry} url)
122 |   where
123 |     printErr : URLParserError -> String
124 |     printErr EmptyString = "Could not parse path, string is empty"
125 |     printErr MissingAuthorityOrPath = "Could not parse path, missing authority or path"

127 | ||| Lookup a value and pop the found key-value pair from the list if it is present
128 | lookupPop : Eq a => List (a, b) -> {default [] acc : List (a, b)} -> a -> Maybe (b, List (a, b))
129 | lookupPop [] v = Nothing
130 | lookupPop ((x, y) :: xs) v
131 |   = if x == v
132 |        then Just (y, reverse acc ++ xs)
133 |        else lookupPop xs {acc = (x, y) :: acc} v

136 | ||| The order of the initial key-value list does not matter but the result is
