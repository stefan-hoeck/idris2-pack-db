 56 | data BufferState = Unread | Read Buffer | Error
 57 |
 58 | wait : HasIO io => IORef BufferState -> io (Maybe Buffer)
 59 | wait buf = case !(readIORef buf) of
 60 |                 Unread => wait buf
 61 |                 Read b => pure (Just b)
 62 |                 Error => pure Nothing

 64 | toString : HasIO io => Buffer -> io String

 66 |   size <- rawSize buf
 67 |   getString buf 0 size

 69 | incToRaw : HasIO io => IncomingMessage -> io (Maybe RawRequest)
 70 | incToRaw inc = do
 71 |   buf <- newIORef (the BufferState Unread)
 72 |   inc.onData (writeIORef buf . Read)
 73 |   inc.onError (\_ => writeIORef buf Error)
 74 |   inc.onEnd (case !(readIORef buf) of -- if the buffer has not been read at the end, alert with Error
 75 |                   Read b => pure ()
 76 |                   _ => writeIORef buf Error)
 77 |   val <- wait buf
 78 |   strVal <- traverse toString val
 79 |   let version = parseVersion inc.httpVersion
 80 |   let method = parseMethod inc.method
 81 |   pure $ mkRequest method inc.url version inc.headers.asList <$> strVal

 84 | fromNodeRequest : IncomingMessage -> RawHttpRequest
 85 | fromNodeRequest nodeReq =
 86 |   let method = parseMethod nodeReq.method
 87 |       path = nodeReq.url
 88 |       headers = nodeReq.headers.asList
 89 |       version = parseVersion nodeReq.httpVersion
 90 |   in mkRequest method path version headers $ MkPublisher $ \s => do
 91 |         nodeReq.onData s.onNext
 92 |         nodeReq.onError s.onFailed
 93 |         nodeReq.onEnd $ s.onSucceded ()
 94 | toNodeResponse : RawHttpResponse -> ServerResponse -> IO ()
 95 | toNodeResponse res nodeRes = do
 96 |   let status = res.status.code
 97 |   headers <- mapHeaders res.headers
 98 |
 99 |   nodeRes.writeHead status headers
100 |   res.body.subscribe $ MkSubscriber
101 |         { onNext = \a => nodeRes.write a Nothing }
102 |         { onFailed = \e => pure () }
103 |         { onSucceded = \_ => nodeRes.end Nothing { d = Buffer } }
104 |   where
105 |     mapHeaders : StringHeaders -> IO Headers
106 |     mapHeaders h = do
107 |       newHeaders <- empty
108 |       foldlM (\hs, (k,v) => hs.setHeader k v) newHeaders h
109 |
110 | computeResponse : HasIO io => (RawHttpRequest -> io RawHttpResponse) -> IncomingMessage -> ServerResponse -> io ()
111 | computeResponse h inc res = do
112 |   let rawReq = fromNodeRequest inc
113 |   rawRes <- h rawReq
114 |   printLn rawReq.method
115 |   printLn rawReq.url
116 |   printLn rawReq.headers
117 |   liftIO $ toNodeResponse rawRes res
